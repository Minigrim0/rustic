7fn check_source_index(idx: usize, system: &System) -> Result<(), String> {
8    if system.sources_len() <= idx {
9        Err(format!(
10            "Source index `{idx}` does not correspond to any source (system contains {} source(s))",
11            system.sources_len()
12        ))
13    } else {
14        Ok(())
15    }
16}
17
18fn check_filter_index<T>(idx: usize, node_map: &[T]) -> Result<(), String> {
19    if node_map.len() <= idx {
20        Err(format!(
21            "Filter index `{idx}` does not correspond to any filter (system contains {} filter(s))",
22            node_map.len()
23        ))
24    } else {
25        Ok(())
26    }
27}

29/// Perform a headless offline render of a `GraphSpec`.
30///
31/// Returns interleaved stereo frames: `Vec<[f32; 2]>` with exactly
32/// `(spec.duration * spec.sample_rate).ceil()` entries.
33pub fn render_graph(spec: &GraphSpec) -> Result<Vec<[f32; 2]>, String> {
34    let mut system = System::new().with_block_size(spec.block_size);
35    let mut source_indices: Vec<usize> = vec![];
36
37    for source in spec.sources.iter() {
38        source_indices.push(system.add_source(build_source(source, spec.sample_rate)));
39    }
40
41    let sink_idx = system.add_sink(Box::new(SimpleSink::new()));
42
43    if spec.filters.is_empty() {
44        for source_index in source_indices.iter() {
45            system.connect_source_to_sink(*source_index, sink_idx);
46        }
47    } else {
48        let nodes = spec
49            .filters
50            .iter()
51            .map(|fs| build_filter(fs, spec.sample_rate).map(|f| system.add_filter(f)))
52            .collect::<Result<Vec<_>, _>>()?;
53
54        for connection in &spec.connections {
55            match connection {
56                crate::spec::ConnectionType::SourceSink { source, sink } => {
57                    check_source_index(*source, &system)?;
58                    system.connect_source_to_sink(*source, *sink);
59                }
60                crate::spec::ConnectionType::FilterSink { filter, sink } => {
61                    check_filter_index(*filter, &nodes)?;
62                    system.connect_sink(nodes[*filter], *sink, 0);
63                }
64                crate::spec::ConnectionType::SourceFilter { source, filter } => {
65                    check_filter_index(*filter, &nodes)?;
66                    check_source_index(*source, &system)?;
67                    system.connect_source(*source, nodes[*filter], 0);
68                }
69                crate::spec::ConnectionType::FilterFilter {
70                    filter_out,
71                    filter_in,
72                } => {
73                    check_filter_index(*filter_out, &nodes)?;
74                    check_filter_index(*filter_in, &nodes)?;
75                    system.connect(nodes[*filter_out], nodes[*filter_in], 0, 0);
76                }
77            }
78        }
79    }
80
81    system
82        .compute()
83        .map_err(|e| format!("Graph compile error: {e:?}"))?;
84
85    let note = Note::from_midi(spec.note);
86    let note_on_block = (spec.note_on * spec.sample_rate / spec.block_size as f32) as usize;
87    let note_off_block = (spec.note_off * spec.sample_rate / spec.block_size as f32) as usize;
88    let total_blocks =
89        (spec.duration * spec.sample_rate / spec.block_size as f32).ceil() as usize + 1;
90
91    let mut frames: Vec<[f32; 2]> = Vec::with_capacity(total_blocks * spec.block_size);
92
93    for i in 0..total_blocks {
94        if i == note_on_block {
95            for source in source_indices.iter() {
96                system.start_note(*source, note, 1.0);
97            }
98        }
99        if i == note_off_block {
100            for source in source_indices.iter() {
101                system.stop_note(*source, note);
102            }
103        }
104        system.run();
105        if let Ok(sink) = system.get_sink(sink_idx) {
106            frames.extend_from_slice(&sink.consume());
107        }
108    }
109
110    let target = (spec.duration * spec.sample_rate).ceil() as usize;
111    frames.truncate(target);
112    Ok(frames)
113}
