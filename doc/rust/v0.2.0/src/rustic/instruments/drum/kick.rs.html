13#[derive(Default, Debug)]
14pub struct Kick {
15    generator: MultiToneGenerator,
16    current_tick: u32,
17    output: f32,
18    playing: bool,
19}
20
21impl Kick {
22    pub fn new() -> Self {
23        Self {
24            generator: MultiToneGeneratorBuilder::new()
25                .add_generator(
26                    ToneGeneratorBuilder::new()
27                        .waveform(Waveform::WhiteNoise)
28                        .frequency_relation(FrequencyRelation::Constant(1.0))
29                        .amplitude_envelope(Box::new(
30                            ADSREnvelopeBuilder::new()
31                                .attack(Box::new(BezierSegment::new(0.0, 0.1, 0.001, (0.0, 1.0))))
32                                .decay(Box::new(LinearSegment::new(0.1, 0.0, 0.1)))
33                                .release(Box::new(ConstantSegment::new(0.0, Some(0.0))))
34                                .build(),
35                        ))
36                        .build(),
37                )
38                .add_generator(
39                    ToneGeneratorBuilder::new()
40                        .waveform(Waveform::Sine)
41                        .frequency_relation(FrequencyRelation::Ratio(1.0))
42                        .amplitude_envelope(Box::new(ConstantSegment::new(1.0, None)))
43                        .build(),
44                )
45                .pitch_envelope(Some(Box::from(BezierSegment::new(
46                    1.0,
47                    0.5,
48                    0.3,
49                    (2.0, 0.2),
50                ))))
51                .amplitude_envelope(Some(Box::new(
52                    ADSREnvelopeBuilder::new()
53                        .attack(Box::new(BezierSegment::new(0.0, 1.0, 0.001, (0.0, 1.0))))
54                        .decay(Box::new(LinearSegment::new(1.0, 0.0, 0.3)))
55                        .release(Box::new(LinearSegment::new(0.0, 0.0, 0.0)))
56                        .build(),
57                )))
58                .mix_mode(MixMode::Sum)
59                .frequency(58.0)
60                .build(),
61            current_tick: 0,
62            output: 0.0,
63            playing: false,
64        }
65    }
66}
67
68impl Instrument for Kick {
69    fn start_note(&mut self, _note: Note, _velocity: f32) {
70        self.current_tick = 0;
71        self.playing = true;
72        self.generator.start();
73    }
74
75    fn stop_note(&mut self, _note: Note) {
76        self.generator.stop();
77    }
78
79    fn get_output(&mut self) -> f32 {
80        self.output
81    }
82
83    fn tick(&mut self) {
84        if !self.playing {
85            self.output = 0.0;
86            return;
87        }
88        self.output = self.generator.tick(1.0 / 44100.0);
89        if self.generator.completed() {
90            self.playing = false;
91        }
92    }
93
94    fn into_system(self: Box<Self>, sample_rate: f32) -> System {
95        let source = MonophonicSource::new_percussive(
96            self.generator,
97            sample_rate,
98            MonophonicAllocationStrategy::Replace,
99        );
100        let mut system = System::new();
101        let source_idx = system.add_source(Box::new(source));
102        let output = system.add_filter(Box::new(GainFilter::new(1.0)));
103        system.connect_source(source_idx, output, 0);
104        let sink_idx = system.add_sink(Box::new(SimpleSink::new()));
105        system.connect_sink(output, sink_idx, 0);
106        system.compute().expect("Kick system compute failed");
107        system
108    }
109}
