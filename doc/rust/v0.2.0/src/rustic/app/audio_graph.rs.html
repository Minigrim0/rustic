1//! AudioGraph — assembles all instruments into a single compiled `System`.
2//!
3//! `AudioGraph::compile()` calls `instrument.into_system()` for each slot,
4//! absorbs every sub-graph into one master graph, and connects all instrument
5//! outputs directly to the `AudioOutputSink`, which handles summing, master
6//! volume, and peak-limiting in one place.
7
8use std::collections::HashMap;
9
10use crate::core::graph::{AudioGraphError, AudioOutputSink, System};
11use crate::instruments::Instrument;
12
13/// A single instrument slot inside the audio graph.
14pub struct InstrumentSlot {
15    /// The instrument that will be converted to a sub-graph on compile.
16    pub instrument: Box<dyn Instrument>,
17    /// Optional per-instrument filter chain (merged after `into_system()`).
18    /// Currently unused — reserved for Phase 4.
19    pub filters: Option<System>,
20}
21
22/// Manages all instruments and compiles them into a single `System`.
23///
24/// After calling `compile()`, `source_map` records which source index inside
25/// the compiled `System` belongs to each instrument slot (by slot index).
26/// Use those indices with `System::start_note()` / `System::stop_note()`.
27#[derive(Default)]
28pub struct AudioGraph {
29    instruments: Vec<InstrumentSlot>,
30    /// Maps slot index → first source index in the most recent compiled System.
31    pub source_map: HashMap<usize, usize>,
32}
33
34impl AudioGraph {
35    pub fn new() -> Self {
36        Self::default()
37    }
38
39    /// Append an instrument and return its slot index.
40    pub fn add_instrument(&mut self, instrument: Box<dyn Instrument>) -> usize {
41        let idx = self.instruments.len();
42        self.instruments.push(InstrumentSlot {
43            instrument,
44            filters: None,
45        });
46        idx
47    }
48
49    /// Number of instrument slots.
50    pub fn len(&self) -> usize {
51        self.instruments.len()
52    }
53
54    pub fn is_empty(&self) -> bool {
55        self.instruments.is_empty()
56    }
57
58    /// Consume all instrument slots and compile them into a unified `System`.
59    ///
60    /// The returned `System` has one `AudioOutputSink` and is ready to be
61    /// swapped into the render thread. The `source_map` is updated so the
62    /// caller can route `NoteStart`/`NoteStop` to the correct source index.
63    ///
64    /// The `instruments` vec is emptied by this call; re-add instruments if
65    /// you need to compile again.
66    pub fn compile(&mut self, sample_rate: f32) -> Result<System, AudioGraphError> {
67        if self.instruments.is_empty() {
68            return Ok(System::silent());
69        }
70
71        let slots: Vec<InstrumentSlot> = std::mem::take(&mut self.instruments);
72        let n = slots.len();
73
74        let mut main = System::new();
75        let mut output_nodes = Vec::with_capacity(n);
76
77        self.source_map.clear();
78
79        for (slot_idx, slot) in slots.into_iter().enumerate() {
80            let source_start = main.sources_len();
81
82            let inst_system = slot.instrument.into_system(sample_rate);
83            let output_node = main.absorb(inst_system)?;
84
85            // Every absorbed source up to source_start is this instrument's
86            let source_count = main.sources_len() - source_start;
87            if source_count > 0 {
88                self.source_map.insert(slot_idx, source_start);
89            }
90
91            output_nodes.push(output_node);
92        }
93
94        // The sink sums all instrument streams, applies master volume and limits.
95        // Each instrument connects directly — no intermediate mixer node needed.
96        let sink = Box::new(AudioOutputSink::new(sample_rate));
97        let sink_idx = main.add_sink(sink);
98        for &out_node in output_nodes.iter() {
99            main.connect_sink(out_node, sink_idx, 0);
100        }
101
102        main.compute()?;
103
104        Ok(main)
105    }
106}
