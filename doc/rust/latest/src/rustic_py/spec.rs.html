41/// Top-level render request from Python.
42#[derive(Debug, Deserialize)]
43pub struct GraphSpec {
44    /// MIDI Note number (0-127)
45    pub note: u8,
46    /// Seconds from t=0 to note-on event.
47    pub note_on: f32,
48    /// Seconds from t=0 to note-off event.
49    pub note_off: f32,
50    /// Total render duration in seconds (must be >= note_off).
51    pub duration: f32,
52    /// Audio sample rate in Hz.
53    #[serde(default = "default_sample_rate")]
54    pub sample_rate: f32,
55    /// Block size for System::run(). Lower = more timing precision, higher = faster.
56    #[serde(default = "default_block_size")]
57    pub block_size: usize,
58    /// A Single mutli-source, with each sub-source having its own parameters
59    pub sources: Vec<MultiSourceSpec>,
60    /// Zero or more filters to be added to the graph
61    #[serde(default)]
62    pub filters: Vec<FilterSpec>,
63    /// All the connections in the graph
64    pub connections: Vec<ConnectionType>,
65}

122/// Frequency relation of a tone relative to the base MIDI note frequency.
123///
124/// Accepted Python forms:
125/// - `"identity"` — same as the played note
126/// - `"harmonic:N"` — N-th harmonic (frequency * N)
127/// - `"ratio:F"` — frequency * F
128/// - `"semitones:I"` — frequency * 2^(I/12)
129/// - `"offset:F"` — frequency + F Hz
130/// - `{"harmonic": N}`, `{"ratio": F}`, `{"semitones": I}`, `{"offset": F}`, `{"constant": F}`
131#[derive(Debug, Deserialize, Default)]
132#[serde(untagged)]
133pub enum FrequencyRelationSpec {
134    #[default]
135    Identity,
136    Tagged(String),
137    Object(HashMap<String, serde_json::Value>),
138}
