13/// Render a synthesis graph spec to stereo audio.
14///
15/// Args:
16///     spec_dict: Python dict conforming to the GraphSpec format.
17///
18/// Returns:
19///     numpy.ndarray of shape (N_samples, 2), dtype float32.
20///
21/// Example::
22///
23/// ```py
24/// import rustic_py
25/// import soundfile as sf
26///
27/// audio = rustic_py.render({
28///     "note": 60, "note_on": 0.0, "note_off": 0.5, "duration": 0.7,
29///     "source": {"waveform": "sine", "attack": 0.01, "decay": 0.1,
30///                "sustain": 0.8, "release": 0.2},
31///     "filters": [{"type": "lowpass", "params": {"cutoff_frequency": 2000.0}}],
32/// })
33/// # audio.shape == (N, 2), dtype float32
34/// sf.write("out.wav", audio, samplerate=44100)
35/// ```
36#[pyfunction]
37fn render<'py>(
38    py: Python<'py>,
39    spec_dict: &Bound<'py, PyDict>,
40) -> PyResult<Bound<'py, PyArray2<f32>>> {
41    let json_module = PyModule::import(py, "json")?;
42    let json_str: String = json_module.call_method1("dumps", (spec_dict,))?.extract()?;
43
44    let spec: GraphSpec = serde_json::from_str(&json_str)
45        .map_err(|e| PyValueError::new_err(format!("Invalid spec: {e}")))?;
46
47    let frames = renderer::render_graph(&spec).map_err(PyValueError::new_err)?;
48
49    let n = frames.len();
50    let flat: Vec<f32> = frames.into_iter().flat_map(|f| f.into_iter()).collect();
51    let array = numpy::ndarray::Array2::from_shape_vec((n, 2), flat)
52        .map_err(|e| PyValueError::new_err(format!("Array shape error: {e}")))?;
53
54    Ok(array.into_pyarray(py))
55}

57/// Returns metadata for all registered filter types.
58///
59/// Returns:
60///     list of dicts, each with keys: name, description, inputs, outputs.
61///
62/// Example::
63///
64/// ```py
65/// import rustic_py
66///
67/// for f in rustic_py.available_filters():
68///     print(f["name"], "-", f["description"])
69/// # lowpass - A simple lowpass filter
70/// # highpass - A simple highpass filter
71/// # ...
72/// ```
73#[pyfunction]
74fn available_filters(py: Python<'_>) -> PyResult<Py<PyAny>> {
75    let filters = rustic::meta::get_filters();
76    let json_str =
77        serde_json::to_string(&filters).map_err(|e| PyValueError::new_err(e.to_string()))?;
78
79    let json_module = PyModule::import(py, "json")?;
80    let obj = json_module.call_method1("loads", (json_str,))?;
81    Ok(obj.unbind())
82}

84/// Returns metadata for all available source (generator/waveform) types.
85///
86/// Returns:
87///     list of dicts, each with keys: name, type_id, description, parameters, output_count.
88///
89/// Example::
90///
91/// ```py
92/// import rustic_py
93///
94/// for s in rustic_py.available_sources():
95///     params = [p["name"] for p in s["parameters"]]
96///     print(f'{s["name"]}: {", ".join(params)}')
97/// # sine: attack, decay, sustain, release
98/// # square: attack, decay, sustain, release
99/// # ...
100/// ```
101#[pyfunction]
102fn available_sources(py: Python<'_>) -> PyResult<Py<PyAny>> {
103    let generators = rustic::meta::get_generators();
104    let json_str =
105        serde_json::to_string(&generators).map_err(|e| PyValueError::new_err(e.to_string()))?;
106
107    let json_module = PyModule::import(py, "json")?;
108    let obj = json_module.call_method1("loads", (json_str,))?;
109    Ok(obj.unbind())
110}

112#[pymodule]
113fn rustic_py(m: &Bound<'_, PyModule>) -> PyResult<()> {
114    m.add_function(wrap_pyfunction!(render, m)?)?;
115    m.add_function(wrap_pyfunction!(available_filters, m)?)?;
116    m.add_function(wrap_pyfunction!(available_sources, m)?)?;
