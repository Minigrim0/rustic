1use rustic::Note;
2use rustic::core::graph::{SimpleSink, System};
3
4use crate::registry::{build_filter, build_source};
5use crate::spec::GraphSpec;
6
7/// Perform a headless offline render of a `GraphSpec`.
8///
9/// Returns interleaved stereo frames: `Vec<[f32; 2]>` with exactly
10/// `(spec.duration * spec.sample_rate).ceil()` entries.
11pub fn render_graph(spec: &GraphSpec) -> Result<Vec<[f32; 2]>, String> {
12    let mut system = System::new().with_block_size(spec.block_size);
13
14    let src_idx = system.add_source(build_source(&spec.source, spec.sample_rate));
15    let sink_idx = system.add_sink(Box::new(SimpleSink::new()));
16
17    if spec.filters.is_empty() {
18        system.connect_source_to_sink(src_idx, sink_idx);
19    } else {
20        let nodes = spec
21            .filters
22            .iter()
23            .map(|fs| build_filter(fs, spec.sample_rate).map(|f| system.add_filter(f)))
24            .collect::<Result<Vec<_>, _>>()?;
25
26        system.connect_source(src_idx, nodes[0], 0);
27        for i in 1..nodes.len() {
28            system.connect(nodes[i - 1], nodes[i], 0, 0);
29        }
30        system.connect_sink(*nodes.last().unwrap(), sink_idx, 0);
31    }
32
33    system
34        .compute()
35        .map_err(|e| format!("Graph compile error: {e:?}"))?;
36
37    let note = Note::from_midi(spec.note);
38    let note_on_block = (spec.note_on * spec.sample_rate / spec.block_size as f32) as usize;
39    let note_off_block = (spec.note_off * spec.sample_rate / spec.block_size as f32) as usize;
40    let total_blocks =
41        (spec.duration * spec.sample_rate / spec.block_size as f32).ceil() as usize + 1;
42
43    let mut frames: Vec<[f32; 2]> = Vec::with_capacity(total_blocks * spec.block_size);
44
45    for i in 0..total_blocks {
46        if i == note_on_block {
47            system.start_note(src_idx, note, 1.0);
48        }
49        if i == note_off_block {
50            system.stop_note(src_idx, note);
51        }
52        system.run();
53        if let Ok(sink) = system.get_sink(sink_idx) {
54            frames.extend_from_slice(&sink.consume());
55        }
56    }
57
58    let target = (spec.duration * spec.sample_rate).ceil() as usize;
59    frames.truncate(target);
60    Ok(frames)
61}
