5/// Represents a Bezier Segment for an envelope. A bezier segment is defined by a starting point,
6/// an ending point and a control point. Since segments times are normalized between 0.0 and 1.0,
7/// the x axis values for the start and end points are, respectively, 0.0 and 1.0.
8///
9/// The control point will define the shape of the curve between the start and end points. A control point below the
10/// line segment formed by the start and end point will cause the curve to bend downwards, similar to an exponential function.
11/// Inversely, a control point above the line segment will cause the curve to bend upwards, similar to a logarithmic function.
12/// A control point on the line segment will cause the curve to be linear.
13pub struct BezierSegment {
14    from: f32,
15    to: f32,
16    duration: f32,
17    control: (f32, f32),
18}
19
20impl BezierSegment {
21    pub fn new(from: f32, to: f32, duration: f32, control_point: (f32, f32)) -> Self {
22        BezierSegment {
23            from,
24            to,
25            duration,
26            control: control_point,
27        }
28    }
29}

37#[typetag::serde]
38impl super::Segment for BezierSegment {
39    /// Evaluates the bezier segment at normalized time `s` ∈ [0, 1].
40    ///
41    /// The control point `(cx, cy)` is in normalized space: `cx` ∈ [0, 1] is the
42    /// *horizontal* position of the control point along the segment, and `cy` is its
43    /// amplitude.  When `cx = 0.5` (the midpoint default) the mapping degenerates to
44    /// a simple quadratic bezier in the amplitude axis only.
45    ///
46    /// For `cx ≠ 0.5` the parametric t is found by inverting the quadratic bezier
47    /// x-component `X(t) = t²(1-2cx) + 2cx·t = s`, which yields:
48    ///   `t = (−cx + √(cx² + s·(1−2cx))) / (1−2cx)`
49    /// The discriminant is always ≥ 0 for `cx, s ∈ [0, 1]`.
50    fn at(&self, time: f32) -> f32 {
51        let s = time.clamp(0.0, 1.0);
52        let cx = self.control.0.clamp(0.0, 1.0);
53        let cy = self.control.1;
54
55        let t = if (cx - 0.5).abs() < 1e-6 {
56            s // linear parametric mapping when cx ≈ 0.5
57        } else {
58            let a = 1.0 - 2.0 * cx;
59            let disc = (cx * cx + s * a).max(0.0).sqrt();
60            ((-cx + disc) / a).clamp(0.0, 1.0)
61        };
62
63        (1.0 - t).powi(2) * self.from + 2.0 * (1.0 - t) * t * cy + t.powi(2) * self.to
64    }
65
66    fn get_duration(&self) -> f32 {
67        self.duration
68    }
69}
70
71#[typetag::serde]
72impl super::super::Envelope for BezierSegment {
73    fn at(&self, time: f32, note_off: f32) -> f32 {
74        if (note_off > 0.0 && time >= note_off) || time >= self.duration {
75            return self.to;
76        }
77        super::Segment::at(self, (time - note_off) / super::Segment::get_duration(self))
78    }
79
80    fn completed(&self, time: f32, note_off: f32) -> bool {
81        (time - note_off) >= super::Segment::get_duration(self)
82    }
83}
