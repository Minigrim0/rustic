4/// Evaluation engine backed by rustic-lang's Session.
5///
6/// Wraps the rustic-lang [`Session`] and converts its output into
7/// display entries for the TUI eval output panel.
8pub struct EvalEngine {
9    session: Session,
10    eval_count: u32,
11}
12
13impl Default for EvalEngine {
14    fn default() -> Self {
15        Self::new()
16    }
17}
18
19impl EvalEngine {
20    pub fn new() -> Self {
21        Self {
22            session: Session::new(),
23            eval_count: 0,
24        }
25    }
26
27    /// Evaluate the given source code and return display entries.
28    pub fn evaluate(&mut self, source: &str) -> Vec<EvalEntry> {
29        self.eval_count += 1;
30        let timestamp = format!("#{:04}", self.eval_count);
31        let mut entries = Vec::new();
32
33        if source.trim().is_empty() {
34            entries.push(EvalEntry {
35                timestamp,
36                kind: EvalEntryKind::Warning,
37                message: "Empty source buffer — nothing to evaluate.".to_string(),
38            });
39            return entries;
40        }
41
42        let result = self.session.evaluate(source);
43
44        // Report parse errors
45        for err in &result.errors {
46            entries.push(EvalEntry {
47                timestamp: timestamp.clone(),
48                kind: EvalEntryKind::Error,
49                message: format!("Line {}: {}", err.location.line, err.message),
50            });
51        }
52
53        // Report deltas
54        for delta in &result.deltas {
55            let (action, name) = match delta {
56                Delta::Add(n) => ("Added", n.as_str()),
57                Delta::Modify(n) => ("Modified", n.as_str()),
58                Delta::Remove(n) => ("Removed", n.as_str()),
59                Delta::Mute(n) => ("Muted", n.as_str()),
60                Delta::Unmute(n) => ("Unmuted", n.as_str()),
61            };
62            entries.push(EvalEntry {
63                timestamp: timestamp.clone(),
64                kind: EvalEntryKind::Info,
65                message: format!("{} pattern: {}", action, name),
66            });
67        }
68
69        // Summary
70        if result.errors.is_empty() {
71            entries.push(EvalEntry {
72                timestamp: timestamp.clone(),
73                kind: EvalEntryKind::Success,
74                message: format!(
75                    "OK — {} active, {} muted ({} changes queued)",
76                    result.patterns_active,
77                    result.patterns_muted,
78                    result.deltas.len(),
79                ),
80            });
81        } else {
82            entries.push(EvalEntry {
83                timestamp: timestamp.clone(),
84                kind: EvalEntryKind::Warning,
85                message: format!(
86                    "{} error(s) — {} active, {} muted",
87                    result.errors.len(),
88                    result.patterns_active,
89                    result.patterns_muted,
90                ),
91            });
92        }
93
94        // Report session state
95        entries.push(EvalEntry {
96            timestamp,
97            kind: EvalEntryKind::Info,
98            message: format!(
99                "Session: bpm={}, sig={}/{}",
100                self.session.bpm, self.session.sig.0, self.session.sig.1
101            ),
102        });
103
104        entries
105    }
106
107    /// Get the current session (for context panel).
108    pub fn _session(&self) -> &Session {
109        &self.session
110    }
111}

113#[cfg(test)]
114mod tests {
115    use super::*;
116
117    #[test]
118    fn test_eval_empty() {
119        let mut engine = EvalEngine::new();
120        let entries = engine.evaluate("");
121        assert_eq!(entries.len(), 1);
122        assert_eq!(entries[0].kind, EvalEntryKind::Warning);
123    }
124
125    #[test]
126    fn test_eval_valid_pattern() {
127        let mut engine = EvalEngine::new();
128        let entries = engine.evaluate("kick kick \"x ~ x ~\"");
129        assert!(entries.iter().any(|e| e.kind == EvalEntryKind::Success));
130        assert!(entries.iter().any(|e| e.message.contains("Added")));
131    }
132
133    #[test]
134    fn test_eval_with_error() {
135        let mut engine = EvalEngine::new();
136        let entries = engine.evaluate("this is broken ???");
137        assert!(entries.iter().any(|e| e.kind == EvalEntryKind::Error));
138    }
139
140    #[test]
141    fn test_eval_detects_changes() {
142        let mut engine = EvalEngine::new();
143        engine.evaluate("kick kick \"x ~ x ~\"");
144        let entries = engine.evaluate("kick kick \"x x x x\"");
145        assert!(entries.iter().any(|e| e.message.contains("Modified")));
146    }
147
148    #[test]
149    fn test_eval_reports_bpm() {
150        let mut engine = EvalEngine::new();
151        let entries = engine.evaluate("bpm 140\nkick kick \"x ~ x ~\"");
152        assert!(entries.iter().any(|e| e.message.contains("bpm=140")));
153    }
