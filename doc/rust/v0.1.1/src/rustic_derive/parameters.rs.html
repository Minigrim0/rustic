6/// Extracts the simple type name from a syn::Type (e.g., "usize", "f32").
7fn type_name(ty: &syn::Type) -> Option<String> {
8    if let syn::Type::Path(path) = ty {
9        path.path.segments.last().map(|s| s.ident.to_string())
10    } else {
11        None
12    }
13}

15fn is_integer_type(ty: &syn::Type) -> bool {
16    type_name(ty).is_some_and(|name| {
17        matches!(
18            name.as_str(),
19            "usize"
20                | "isize"
21                | "u8"
22                | "u16"
23                | "u32"
24                | "u64"
25                | "u128"
26                | "i8"
27                | "i16"
28                | "i32"
29                | "i64"
30                | "i128"
31        )
32    })
33}

35/// Extracts the range parameter from the token stream.
36fn extract_range_parameter(
37    field_name: String,
38    title: String,
39    stream: TokenStream,
40) -> Parameter<String> {
41    let mut values = stream.into_iter();
42    let minimum = values
43        .next()
44        .unwrap_or_else(|| panic!("No minimum value found"));
45    let minimum = if let TokenTree::Punct(e) = minimum {
46        let minimum = values
47            .next()
48            .unwrap_or_else(|| panic!("No minimum value found"));
49
50        let minimum: f32 = if let TokenTree::Literal(m) = minimum {
51            m.to_string().parse().unwrap()
52        } else {
53            panic!("Minimum value is not a literal: found {:?}", minimum);
54        };
55
56        match e.as_char() {
57            '-' => -minimum,
58            '+' => minimum,
59            _ => panic!(
60                "Unable to parse minimum value, Punct must be in [+, -] but found {:?}",
61                e
62            ),
63        }
64    } else if matches!(minimum, TokenTree::Literal(_)) {
65        if let TokenTree::Literal(m) = minimum {
66            m.to_string().parse().unwrap()
67        } else {
68            panic!("Minimum value is not a literal: found {:?}", minimum);
69        }
70    } else {
71        panic!("Expected literal or Punct for minimum value");
72    };
73
74    let maximum = values
75        .next()
76        .unwrap_or_else(|| panic!("No maximum value found"));
77    let default = values
78        .next()
79        .unwrap_or_else(|| panic!("No default value found"));
80
81    let maximum = if let TokenTree::Literal(maximum) = maximum {
82        maximum.to_string().parse().unwrap()
83    } else {
84        panic!("Maximum value is not a literal");
85    };
86
87    let default = if let TokenTree::Literal(default) = default {
88        default.to_string().parse().unwrap()
89    } else {
90        panic!("Default value is not a literal");
91    };
92
93    Parameter::Range {
94        title,
95        field_name,
96        min: minimum,
97        max: maximum,
98        default,
99        value: default,
100    }
101}

103/// Extracts the toggle parameter from the token stream.
104fn extract_toggle_parameter(
105    field_name: String,
106    title: String,
107    stream: TokenStream,
108) -> Parameter<String> {
109    let mut values = stream.into_iter();
110    if let Some(value) = values.next() {
111        let value = if let TokenTree::Literal(value) = value {
112            value.to_string().parse().unwrap()
113        } else {
114            panic!("Value is not a literal");
115        };
116
117        Parameter::Toggle {
118            title,
119            field_name,
120            value,
121            default: value,
122        }
123    } else {
124        panic!("No value found for toggle parameter");
125    }
126}

128/// Extracts the float parameter from the token stream.
129/// A float parameter is an unbounded parameter (contrary to
130/// a range parameter).
131fn extract_float_parameter(
132    field_name: String,
133    title: String,
134    stream: TokenStream,
135) -> Parameter<String> {
136    let value = stream.into_iter().next().unwrap();
137    let value = if let TokenTree::Literal(value) = value {
138        value.to_string().parse().unwrap()
139    } else {
140        panic!("Value is not a literal");
141    };
142
143    Parameter::Float {
144        title,
145        field_name,
146        default: value,
147        value,
148    }
149}

151/// Extracts the int parameter from the token stream.
152/// Syntax: `filter_parameter(int, <default>)` or
153///         `filter_parameter(int, <default>, <min>)` or
154///         `filter_parameter(int, <default>, <min>, <max>)`
155fn extract_int_parameter(
156    field_name: String,
157    title: String,
158    stream: TokenStream,
159) -> Parameter<String> {
160    let values: Vec<TokenTree> = stream.into_iter().collect();
161
162    let default: i32 = values
163        .first()
164        .map(|t| {
165            t.to_string()
166                .parse()
167                .expect("Expected an integer default value")
168        })
169        .expect("No default value found for int parameter");
170
171    let min: Option<i32> = values.get(1).map(|t| {
172        t.to_string()
173            .parse()
174            .expect("Expected an integer min value")
175    });
176
177    let max: Option<i32> = values.get(2).map(|t| {
178        t.to_string()
179            .parse()
180            .expect("Expected an integer max value")
181    });
182
183    Parameter::Int {
184        title,
185        field_name,
186        default,
187        value: default,
188        min,
189        max,
190    }
191}

193/// Returns the natural bounds (min, max) for bounded integer types.
194/// Types like `usize`, `isize`, `i64`, `u64` are unbounded (too large for i32 metadata).
195fn type_bounds(ty: &syn::Type) -> (Option<i32>, Option<i32>) {
196    match type_name(ty).as_deref() {
197        Some("u8") => (Some(0), Some(u8::MAX as i32)),
198        Some("i8") => (Some(i8::MIN as i32), Some(i8::MAX as i32)),
199        Some("u16") => (Some(0), Some(u16::MAX as i32)),
200        Some("i16") => (Some(i16::MIN as i32), Some(i16::MAX as i32)),
201        // u32/i32 max fits in i32 only for i32; u32::MAX overflows i32
202        Some("u32") => (Some(0), None),
203        Some("usize") => (Some(0), None),
204        // Signed types without useful i32-representable bounds
205        _ => (None, None),
206    }
207}

209/// Extracts a `val` parameter by inferring the type from the field.
210/// Integer-like fields produce `Parameter::Int`, float-like fields produce `Parameter::Float`.
211/// Bounded types (u8, i8, u16, i16, etc.) automatically get their natural bounds
212/// when none are explicitly specified.
213/// Syntax: `filter_parameter(val, <default> [, <min> [, <max>]])`
214fn extract_val_parameter(
215    field_name: String,
216    title: String,
217    field_type: &syn::Type,
218    stream: TokenStream,
219) -> Parameter<String> {
220    if is_integer_type(field_type) {
221        let mut param = extract_int_parameter(field_name, title, stream);
222        // Fill in natural type bounds where the user didn't specify them
223        let (type_min, type_max) = type_bounds(field_type);
224        if let Parameter::Int {
225            ref mut min,
226            ref mut max,
227            ..
228        } = param
229        {
230            if min.is_none() {
231                *min = type_min;
232            }
233            if max.is_none() {
234                *max = type_max;
235            }
236        }
237        param
238    } else {
239        // Float-like types
240        extract_float_parameter(field_name, title, stream)
241    }
242}

244pub fn extract_vector_parameter(
245    field_name: String,
246    title: String,
247    stream: TokenStream,
248) -> Parameter<String> {
249    let mut values = stream.into_iter();
250
251    // First token: either a field name (Ident) or a constant size (Literal)
252    let size = match values.next() {
253        Some(TokenTree::Ident(ident)) => rustic_meta::ListSize::Field(ident.to_string()),
254        Some(TokenTree::Literal(lit)) => {
255            let n: usize = lit
256                .to_string()
257                .parse()
258                .expect("Expected a positive integer for the constant list size");
259            rustic_meta::ListSize::Constant(n)
260        }
261        _ => panic!(
262            "Expected an identifier (field name) or integer literal (constant size) for list size"
263        ),
264    };
265
266    // Second token: the element type (float, range, toggle)
267    let ltype_name = if let Some(TokenTree::Ident(ident)) = values.next() {
268        ident.to_string()
269    } else {
270        panic!("Expected an identifier for the element type");
271    };
272
273    // Remaining tokens: parameters for the element type
274    let remaining: Vec<TokenTree> = values.collect();
275
276    let element_title = title.clone();
277    let ltype = match ltype_name.as_str() {
278        "float" => {
279            let default: f32 = remaining
280                .first()
281                .map(|t| {
282                    t.to_string()
283                        .parse()
284                        .expect("Expected a float default value")
285                })
286                .unwrap_or(0.0);
287            rustic_meta::Literal::Float(element_title, default)
288        }
289        "range" => {
290            let min: f32 = remaining
291                .first()
292                .map(|t| t.to_string().parse().expect("Expected a float min value"))
293                .unwrap_or(0.0);
294            let max: f32 = remaining
295                .get(1)
296                .map(|t| t.to_string().parse().expect("Expected a float max value"))
297                .unwrap_or(1.0);
298            let default: f32 = remaining
299                .get(2)
300                .map(|t| {
301                    t.to_string()
302                        .parse()
303                        .expect("Expected a float default value")
304                })
305                .unwrap_or(0.0);
306            rustic_meta::Literal::Range(element_title, min, max, default)
307        }
308        "toggle" => {
309            let default: bool = remaining
310                .first()
311                .map(|t| {
312                    t.to_string()
313                        .parse()
314                        .expect("Expected a bool default value")
315                })
316                .unwrap_or(false);
317            rustic_meta::Literal::Toggle(element_title, default)
318        }
319        "int" => {
320            let default: i32 = remaining
321                .first()
322                .map(|t| {
323                    t.to_string()
324                        .parse()
325                        .expect("Expected an int default value")
326                })
327                .unwrap_or(0);
328            let min: Option<i32> = remaining
329                .get(1)
330                .map(|t| t.to_string().parse().expect("Expected an int min value"));
331            let max: Option<i32> = remaining
332                .get(2)
333                .map(|t| t.to_string().parse().expect("Expected an int max value"));
334            rustic_meta::Literal::Int(element_title, default, min, max)
335        }
336        other => panic!("Unknown element type for list parameter: {other}"),
337    };
338
339    Parameter::List {
340        title,
341        field_name,
342        size,
343        ltype,
344    }
345}

347/// Iterates through the token stream of a filter_parameter
348/// attribute to extract the parameter name and any additional
349/// information.
350pub fn extract_parameter(
351    name: String,
352    field_type: &syn::Type,
353    token_stream: TokenStream,
354) -> Parameter<String> {
355    let mut stream = token_stream.into_iter();
356    let param_type = if let Some(TokenTree::Ident(token)) = stream.next() {
357        token.to_string()
358    } else {
359        panic!("No parameter type found");
360    };
361
362    let values = stream
363        .filter(|e| {
364            if let TokenTree::Punct(punct) = e {
365                punct.as_char() != ','
366            } else {
367                true
368            }
369        })
370        .collect();
371
372    let parameter_title: String = name.to_case(Case::Title);
373
374    match param_type.as_str() {
375        "range" => extract_range_parameter(name, parameter_title, values),
376        "toggle" => extract_toggle_parameter(name, parameter_title, values),
377        "float" => extract_float_parameter(name, parameter_title, values),
378        "int" => extract_int_parameter(name, parameter_title, values),
379        "val" => extract_val_parameter(name, parameter_title, field_type, values),
380        "list" | "vec" => extract_vector_parameter(name, parameter_title, values),
381        any => panic!("Unknown parameter type: {any}"),
382    }
383}
