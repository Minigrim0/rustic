38/// ## A Pipe & Filter system
39/// The system is composed of filters, sources and sinks.
40/// It is represented as a directed graph where the filters are the nodes.
41/// The Sources & Sinks are special nodes that have respectively only outgoing (source) or incoming (sink) edges.
42/// The edges represent the pipes between the filters.
43/// Some filters have special output properties. E.g. the delay filter's input pipe is ignored (postponed) when
44/// the topology sorting is done, in order to avoid cycles. A system with cycles must include a delay or similar filter
45/// to break the cycle.
46///
47/// ```rust
48/// use rustic::core::graph::System;
49/// use rustic::core::filters::prelude::Tremolo;
50///
51/// // A simple system with one input and one output
52/// let mut system = System::new();
53///
54/// // Adding a filter to the system
55/// let filter = Tremolo::new(20.0, 0.5, 44100.0);
56/// let filter_index = system.add_filter(Box::from(filter));
57/// ```
58#[derive(Debug, Clone)]
59#[allow(clippy::type_complexity)]
60pub struct System {
61    // The actual filter graph, from which the execution order is derived
62    // Each weight represents the port into which the filter is connected
63    graph: Graph<AudioNode, (usize, usize)>,
64    // Each layer represents filters that can be run concurrently.
65    layers: Vec<Vec<usize>>,
66    // The sources of the system and the filters they are connected to.
67    // Each source may fan out to multiple (filter, port) pairs.
68    sources: Vec<(Box<dyn Source>, Vec<(NodeIndex<u32>, usize)>)>,
69    // The sinks of the system.
70    // The node index is the index of the filter that the sink is connected to
71    // The second usize is the port of the filter that the sink is connected to
72    sinks: Vec<((NodeIndex<u32>, usize), Box<dyn Sink>)>,
73    /// Direct source→sink wires that bypass the filter graph entirely.
74    /// Each entry is (source_index, sink_index).
75    source_sink_wires: Vec<(usize, usize)>,
76    /// Live modulation wires: a source's block-mean drives a named parameter.
77    mod_wires: Vec<ModWire>,
78    /// Number of frames to produce per `run()` call
79    block_size: usize,
80}

128    /// Merges the two systems to create a new one. The graphs are merged following the given mapping from sinks to sources.
129    /// Sinks to sources links are replaced with a simple combinator filter. The amount of input in the second system
130    /// should match the amount of output in the first system.
131    #[allow(clippy::type_complexity)]
132    pub fn merge(
133        mut self,
134        other: System,
135        mapping: Vec<(usize, usize)>,
136    ) -> Result<System, AudioGraphError> {
137        if self.sinks.len() != other.sinks.len() {
138            log::error!("Trying to merge graphs with incompatible shapes");
139            return Err(AudioGraphError::InvalidMerging);
140        }
141
142        // Contains the mapping other graph -> new graph
143        let mut new_edge_map: HashMap<NodeIndex, NodeIndex> = HashMap::new();
144
145        log::trace!("Merging two graphs together");
146        for (from, to) in mapping.iter() {
147            let (graph_b_source_descendant_index, graph_b_node_port) = other.sources[*to]
148                .1
149                .first()
150                .copied()
151                .unwrap_or((NodeIndex::new(0), 0));
152            let source_descendant = other.graph[graph_b_source_descendant_index].clone();
153
154            // Save the new index of the source descendant
155            let new_index = if let std::collections::hash_map::Entry::Vacant(e) =
156                new_edge_map.entry(graph_b_source_descendant_index)
157            {
158                let new_index = self.graph.add_node(source_descendant);
159                log::info!(
160                    "idx {} -> idx {}",
161                    graph_b_source_descendant_index.index(),
162                    new_index.index()
163                );
164                e.insert(new_index);
165                new_index
166            } else {
167                match new_edge_map.get(&graph_b_source_descendant_index) {
168                    Some(v) => *v,
169                    None => panic!("What ?"),
170                }
171            };
172
173            // Connect the sink's predecessors to the source's successors
174            let (sink_predecessor_id, sink_predecessor_port) = self.sinks[*from].0;
175            log::trace!(
176                "\tNode {} -> Sink {} & Source {} -> Node {} => Node {} -> Node {}",
177                sink_predecessor_id.index(),
178                from,
179                to,
180                graph_b_source_descendant_index.index(),
181                sink_predecessor_id.index(),
182                new_index.index()
183            );
184            self.graph.add_edge(
185                sink_predecessor_id,
186                new_index,
187                (sink_predecessor_port, graph_b_node_port),
188            );
189        }
190
191        // Go through all nodes in the other graph and add them to the new graph
192        for node_index in other.graph.node_indices() {
193            // Skip already added nodes
194            if new_edge_map.contains_key(&node_index) {
195                continue;
196            }
197
198            let node = other.graph[node_index].clone();
199            let new_index = self.graph.add_node(node);
200            new_edge_map.insert(node_index, new_index);
201        }
202
203        // Now that all nodes are added, connect the edges
204        for edge in other.graph.edge_indices() {
205            let (other_from, other_to) = other.graph.edge_endpoints(edge).unwrap();
206            let (from, to) = (new_edge_map[&other_from], new_edge_map[&other_to]);
207            log::info!(
208                "\tEdge ({}, {}) -> ({}, {})",
209                other_from.index(),
210                other_to.index(),
211                from.index(),
212                to.index()
213            );
214            let weight = other.graph[edge];
215            self.graph.add_edge(from, to, weight);
216        }
217
218        let new_sinks: Vec<_> = other
219            .sinks
220            .iter()
221            .map(|sink| {
222                (
223                    (new_edge_map[&sink.0.0], sink.0.1),
224                    dyn_clone::clone_box(&*sink.1),
225                )
226            })
227            .collect();
228
229        let new_system: System = System {
230            graph: self.graph,
231            layers: self.layers,
232            sources: self.sources,
233            sinks: new_sinks,
234            source_sink_wires: Vec::new(),
235            mod_wires: Vec::new(),
236            block_size: self.block_size,
237        };
238
239        Ok(new_system)
240    }

354    /// Removes a source by index
355    pub fn remove_source(&mut self, index: usize) -> Option<Box<dyn Source>> {
356        if index < self.sources.len() {
357            let removed = self.sources.remove(index).0;
358            // Drop direct source→sink wires; shift remaining source indices
359            self.source_sink_wires.retain(|&(s, _)| s != index);
360            for (s, _) in self.source_sink_wires.iter_mut() {
361                if *s > index {
362                    *s -= 1;
363                }
364            }
365            // Drop mod wires involving the removed source; reindex surviving entries
366            self.mod_wires.retain(|w| w.from_source != index);
367            for w in self.mod_wires.iter_mut() {
368                if w.from_source > index {
369                    w.from_source -= 1;
370                }
371                if let ModTarget::Source(ref mut t) = w.target
372                    && *t > index
373                {
374                    *t -= 1;
375                }
376            }
377            Some(removed)
378        } else {
379            None
380        }
381    }

448    // Creates the execution layers by sorting the graph topologically.
449    #[allow(clippy::result_unit_err)]
450    pub fn compute(&mut self) -> Result<(), AudioGraphError> {
451        self.layers.clear();
452
453        // Makes the graph acyclic to be able to create a topology sort
454        let acyclic_graph = self.graph.filter_map(
455            |_index, node| Some(node),
456            |index, edge| {
457                if self
458                    .graph
459                    .edge_endpoints(index)
460                    .map(|(_, to)| self.graph[to].postponable())
461                    == Some(true)
462                {
463                    None
464                } else {
465                    Some(edge)
466                }
467            },
468        );
469
470        let topo = toposort(&acyclic_graph, None).map_err(|_| AudioGraphError::CycleDetected)?;
471        for node in topo {
472            // TODO: Add same-layer ability (to run some filters in parallel)
473            self.layers.push(vec![node.index()])
474        }
475
476        Ok(())
477    }

479    // Performs one full run of the system, running every filter once in an order such that data
480    // that entered the system this run can exit it this run as well.
481    pub fn run(&mut self) {
482        let block_size = self.block_size;
483
484        // Pull from all sources; push directly to connected AudioNodes (fan-out).
485        // Two-step collect releases the borrow on self.sources before we touch self.graph.
486        let source_blocks: Vec<Arc<Block>> = self
487            .sources
488            .iter_mut()
489            .enumerate()
490            .map(|(i, (source, connections))| {
491                let block = source.pull(block_size);
492                log::trace!("[system::run] source[{i}] active={}", source.is_active());
493                (block, connections)
494            })
495            .collect::<Vec<_>>()
496            .into_iter()
497            .map(|(block, connections)| {
498                let block = Arc::from(block);
499                for (desc, port) in connections {
500                    if let Some(node) = self.graph.node_weight_mut(*desc) {
501                        node.push(Arc::clone(&block), *port);
502                    }
503                }
504                block
505            })
506            .collect();
507
508        // Apply live modulation: drive target parameters from source block means.
509        let mod_actions: Vec<(ModTarget, String, f32)> = self
510            .mod_wires
511            .iter()
512            .filter_map(|wire| {
513                let block = source_blocks.get(wire.from_source)?;
514                let mean = if block.is_empty() {
515                    0.0
516                } else {
517                    block.iter().map(|f| (f[0] + f[1]) * 0.5).sum::<f32>() / block.len() as f32
518                };
519                Some((wire.target.clone(), wire.param_name.clone(), mean))
520            })
521            .collect();
522        for (target, param_name, value) in mod_actions {
523            match target {
524                ModTarget::Source(idx) => {
525                    if let Some((src, _)) = self.sources.get_mut(idx) {
526                        src.set_parameter(&param_name, value);
527                    }
528                }
529                ModTarget::Filter(node_idx) => {
530                    if let Some(node) = self.graph.node_weight_mut(node_idx) {
531                        node.filter_mut().set_parameter(&param_name, value);
532                    }
533                }
534            }
535        }
536
537        // Process filters layer by layer.
538        for layer in self.layers.iter() {
539            for &f in layer.iter() {
540                let node_idx = NodeIndex::new(f);
541
542                // Process: mix accumulated inputs, run transform, get Arc-wrapped outputs.
543                let outputs = self.graph[node_idx].process(block_size);
544
545                // Push outputs to downstream nodes.
546                let neighbours: Vec<NodeIndex> = self
547                    .graph
548                    .neighbors_directed(node_idx, Direction::Outgoing)
549                    .collect();
550                for neighbour in neighbours {
551                    let edges: Vec<(usize, usize)> = self
552                        .graph
553                        .edges_connecting(node_idx, neighbour)
554                        .map(|e| *e.weight())
555                        .collect();
556                    for (out_port, in_port) in edges {
557                        if let Some(block) = outputs.get(out_port)
558                            && let Some(node) = self.graph.node_weight_mut(neighbour)
559                        {
560                            node.push(Arc::clone(block), in_port);
561                        }
562                    }
563                }
564
565                // Push outputs to connected sinks inline (no second pass needed).
566                for ((sink_node, sink_port), sink) in &mut self.sinks {
567                    if *sink_node == node_idx
568                        && let Some(block) = outputs.get(*sink_port)
569                    {
570                        log::trace!(
571                            "[system::run] sink ← NodeIndex({}) port={sink_port}",
572                            node_idx.index()
573                        );
574                        sink.push(Arc::clone(block), 0);
575                    }
576                }
577            }
578        }
579
580        // Push source blocks to directly-wired sinks.
581        for &(src_idx, sink_idx) in &self.source_sink_wires {
582            if let Some(block) = source_blocks.get(src_idx)
583                && let Some((_, sink)) = self.sinks.get_mut(sink_idx)
584            {
585                sink.push(Arc::clone(block), 0);
586            }
587        }
588    }

657    pub fn get_filter_mut(&mut self, index: NodeIndex<u32>) -> Option<&mut Box<dyn Filter>> {
658        self.graph.node_weight_mut(index).map(|n| n.filter_mut())
659    }

680    /// Absorbs all filter nodes, edges, and sources from `other` into `self`,
681    /// remapping `NodeIndex`es to the new graph.
682    ///
683    /// Returns the remapped `NodeIndex` of the filter that was feeding `other`'s
684    /// first sink — the "output node" of the absorbed sub-graph — so the caller
685    /// can wire it into a master combinator or sink.
686    ///
687    /// `other`'s sinks are intentionally **not** imported; the caller is responsible
688    /// for providing a master sink and connecting to the returned output node.
689    pub fn absorb(&mut self, other: System) -> Result<NodeIndex<u32>, AudioGraphError> {
690        if other.sinks.is_empty() {
691            return Err(AudioGraphError::InvalidMerging);
692        }
693
694        // Record which filter node fed other's first sink before we consume other
695        let (other_output_node, _) = other.sinks[0].0;
696
697        // Import all filter nodes, building old-index → new-index mapping
698        let mut remap: HashMap<NodeIndex<u32>, NodeIndex<u32>> = HashMap::new();
699        for old_idx in other.graph.node_indices() {
700            let node = other.graph[old_idx].clone();
701            let new_idx = self.graph.add_node(node);
702            remap.insert(old_idx, new_idx);
703        }
704
705        // Re-add edges with remapped endpoints
706        for edge_idx in other.graph.edge_indices() {
707            let (from, to) = other.graph.edge_endpoints(edge_idx).unwrap();
708            let weight = other.graph[edge_idx];
709            self.graph.add_edge(remap[&from], remap[&to], weight);
710        }
711
712        // Transfer sources, remapping their connected filter NodeIndexes
713        for (source, connections) in other.sources {
714            let remapped: Vec<(NodeIndex<u32>, usize)> = connections
715                .into_iter()
716                .map(|(node_idx, port)| {
717                    let remapped_idx = remap.get(&node_idx).copied().unwrap_or(node_idx);
718                    (remapped_idx, port)
719                })
720                .collect();
721            self.sources.push((source, remapped));
722        }
723
724        // Return the remapped output node so the caller can connect it
725        remap
726            .get(&other_output_node)
727            .copied()
728            .ok_or(AudioGraphError::InvalidNode)
729    }
