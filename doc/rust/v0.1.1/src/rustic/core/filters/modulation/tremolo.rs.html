1use std::fmt;
2use std::sync::Arc;
3
4use rustic_derive::FilterMetaData;
5
6use crate::core::Block;
7use crate::core::graph::{Entry, Filter};
8
9/// A Tremolo filter, that changes sound amplitude on a sinusoid
10/// basis.
11#[derive(FilterMetaData, Debug, Clone, Default)]
12pub struct Tremolo {
13    #[filter_source]
14    source: Arc<Block>,
15    phase: f32,
16    #[filter_parameter(range, 0.0, 20.0, 1.0)]
17    pub frequency: f32,
18    #[filter_parameter(range, 0.0, 1.0, 0.5)]
19    pub depth: f32,
20    #[filter_parameter(range, 0.0, 192000.0, 44100.0)]
21    pub sample_rate: f32,
22}
23
24impl Tremolo {
25    pub fn new(frequency: f32, depth: f32, sample_rate: f32) -> Self {
26        Self {
27            source: Arc::new(Vec::new()),
28            phase: 0.0,
29            frequency,
30            depth,
31            sample_rate,
32        }
33    }
34}
35
36impl fmt::Display for Tremolo {
37    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
38        write!(f, "Tremolo: {}Hz, depth: {}", self.frequency, self.depth)
39    }
40}
41
42impl Entry for Tremolo {
43    fn push(&mut self, block: Arc<Block>, _port: usize) {
44        self.source = block;
45    }
46}
47
48impl Filter for Tremolo {
49    fn transform(&mut self) -> Vec<Block> {
50        let phase_increment =
51            (2.0 * std::f32::consts::PI * self.frequency) / self.sample_rate.max(1.0);
52
53        let output: Block = self
54            .source
55            .iter()
56            .map(|frame| {
57                let modulation = 1.0 - self.depth * (0.5 * (1.0 + self.phase.sin()));
58                self.phase += phase_increment;
59                if self.phase > 2.0 * std::f32::consts::PI {
60                    self.phase -= 2.0 * std::f32::consts::PI;
61                }
62                std::array::from_fn(|ch| frame[ch] * modulation)
63            })
64            .collect();
65
66        vec![output]
67    }
68
69    fn as_any_mut(&mut self) -> &mut dyn std::any::Any {
70        self
71    }
72}
