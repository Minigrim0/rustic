15#[cfg_attr(mobile, tauri::mobile_entry_point)]
16pub fn run() {
17    colog::init();
18    log::set_max_level(log::LevelFilter::Info);
19
20    info!("Starting Rustic Sample Analyser");
21
22    tauri::Builder::default()
23        .plugin(tauri_plugin_opener::init())
24        .plugin(tauri_plugin_fs::init())
25        .manage(RwLock::new(state::AudioState::default()))
26        .invoke_handler(tauri::generate_handler![
27            commands::analyze::analyze_audio_file,
28            commands::query::get_waveform,
29            commands::query::get_spectrum,
30            commands::query::get_top_frequencies,
31            commands::query::get_spectrogram,
32            commands::utils::frequency_to_note_command,
33            commands::utils::save_analysis,
34        ])
35        .setup(|app| {
36            // File system scope
37            let scope = app.fs_scope();
38            scope.allow_directory("/tmp", true)?;
39
40            // Native menu
41            let file_menu = SubmenuBuilder::new(app, "File")
42                .item(
43                    &MenuItemBuilder::with_id("open", "Open File")
44                        .accelerator("CmdOrCtrl+O")
45                        .build(app)?,
46                )
47                .separator()
48                .item(&PredefinedMenuItem::quit(app, Some("Quit"))?)
49                .build()?;
50
51            let help_menu = SubmenuBuilder::new(app, "Help")
52                .item(&MenuItemBuilder::with_id("about", "About Rustic").build(app)?)
53                .build()?;
54
55            let menu = MenuBuilder::new(app)
56                .items(&[&file_menu, &help_menu])
57                .build()?;
58
59            app.set_menu(menu)?;
60
61            app.on_menu_event(|app, event| {
62                let _ = app.emit(event.id().as_ref(), ());
63            });
64
65            Ok(())
66        })
67        .run(tauri::generate_context!())
68        .expect("error while running tauri application");
69}
