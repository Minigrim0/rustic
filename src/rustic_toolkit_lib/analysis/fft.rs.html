7#[derive(Debug, Clone, Serialize, Deserialize, TS)]
8#[ts(export, export_to = "../../src/types/")]
9pub struct FrequencyData {
10    pub frequency: f32,
11    pub magnitude: f32,
12    pub phase: f32,
13}
14
15/// Computes the Fast Fourier Transform for the given samples
16pub fn compute_fft(samples: &[f32], sample_rate: u32) -> Vec<FrequencyData> {
17    info!(
18        "Computing FFT for {} samples at {} Hz",
19        samples.len(),
20        sample_rate
21    );
22
23    // We need a power of 2 for the FFT size
24    let mut fft_size = 1;
25    while fft_size < samples.len() {
26        fft_size *= 2;
27    }
28
29    // Prepare input data for FFT
30    let mut fft_input: Vec<Complex<f32>> = Vec::with_capacity(fft_size);
31
32    // Apply Hann window function to reduce spectral leakage
33    for (i, &sample) in samples.iter().take(fft_size).enumerate() {
34        let window_val = 0.5 * (1.0 - (2.0 * PI * i as f32 / (fft_size as f32 - 1.0)).cos());
35        fft_input.push(Complex {
36            re: sample * window_val,
37            im: 0.0,
38        });
39    }
40
41    // Pad with zeros if needed
42    fft_input.resize(fft_size, Complex { re: 0.0, im: 0.0 });
43
44    // Create FFT planner
45    let mut planner = FftPlanner::new();
46    let fft = planner.plan_fft_forward(fft_size);
47
48    // Perform FFT
49    let mut fft_output = fft_input;
50    fft.process(&mut fft_output);
51
52    // Convert FFT output to frequency data
53    let nyquist = fft_size / 2;
54    let mut result = Vec::with_capacity(nyquist);
55
56    for (i, complex) in fft_output.iter().take(nyquist).enumerate() {
57        let frequency = i as f32 * sample_rate as f32 / fft_size as f32;
58        let magnitude = (complex.re.powi(2) + complex.im.powi(2)).sqrt();
59        let phase = complex.im.atan2(complex.re);
60
61        result.push(FrequencyData {
62            frequency,
63            magnitude,
64            phase,
65        });
66    }
67
68    // Normalize magnitudes
69    if let Some(max_magnitude) = result.iter().map(|f| f.magnitude).reduce(f32::max)
70        && max_magnitude > 0.0
71    {
72        for freq_data in &mut result {
73            freq_data.magnitude /= max_magnitude;
74        }
75    }
76
77    info!("FFT completed: {} frequency bins generated", result.len());
78    result
79}

81#[cfg(test)]
82mod tests {
83    use super::*;
84
85    #[test]
86    fn test_compute_fft_sine_wave() {
87        // Create a sine wave at 440 Hz
88        let sample_rate = 44100;
89        let frequency = 440.0;
90        let duration = 0.1; // 100ms
91        let num_samples = (sample_rate as f32 * duration) as usize;
92
93        let mut samples = Vec::with_capacity(num_samples);
94        for i in 0..num_samples {
95            let t = i as f32 / sample_rate as f32;
96            samples.push((2.0 * PI * frequency * t).sin());
97        }
98
99        let frequencies = compute_fft(&samples, sample_rate);
100
101        // Find the peak frequency
102        let peak = frequencies
103            .iter()
104            .max_by(|a, b| a.magnitude.partial_cmp(&b.magnitude).unwrap())
105            .unwrap();
106
107        // The peak frequency should be close to 440 Hz
108        assert!((peak.frequency - 440.0).abs() < 10.0);
109    }
