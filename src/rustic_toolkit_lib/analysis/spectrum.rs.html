5/// Computes a time-frequency spectrogram for the given samples
6pub fn compute_spectrum(samples: &[f32], sample_rate: u32) -> Vec<Vec<f32>> {
7    info!(
8        "Computing spectrum for {} samples at {} Hz",
9        samples.len(),
10        sample_rate
11    );
12
13    // Parameters for the Short-Time Fourier Transform (STFT)
14    let window_size = 1024;
15    let hop_size = window_size / 4; // 75% overlap
16    let num_frames = samples.len().saturating_sub(window_size) / hop_size + 1;
17
18    // Create the output spectrogram
19    let mut spectrogram = Vec::with_capacity(num_frames);
20
21    // Create FFT planner
22    let mut planner = FftPlanner::new();
23    let fft = planner.plan_fft_forward(window_size);
24
25    // Process each frame
26    for frame_index in 0..num_frames {
27        let start_index = frame_index * hop_size;
28
29        // Apply window function to the frame
30        let mut fft_input: Vec<Complex<f32>> = Vec::with_capacity(window_size);
31        for i in 0..window_size {
32            if start_index + i < samples.len() {
33                let window_val =
34                    0.5 * (1.0 - (2.0 * PI * i as f32 / (window_size as f32 - 1.0)).cos());
35                fft_input.push(Complex {
36                    re: samples[start_index + i] * window_val,
37                    im: 0.0,
38                });
39            } else {
40                fft_input.push(Complex { re: 0.0, im: 0.0 });
41            }
42        }
43
44        // Perform FFT
45        let mut fft_output = fft_input;
46        fft.process(&mut fft_output);
47
48        // Convert FFT output to magnitudes (only using half due to Nyquist)
49        let nyquist = window_size / 2;
50        let mut frame_magnitudes = Vec::with_capacity(nyquist);
51
52        for complex in fft_output.iter().take(nyquist) {
53            let magnitude = (complex.re.powi(2) + complex.im.powi(2)).sqrt();
54            frame_magnitudes.push(magnitude);
55        }
56
57        // Add frame to spectrogram
58        spectrogram.push(frame_magnitudes);
59    }
60
61    info!(
62        "Spectrogram completed: {} frames, {} frequency bins",
63        spectrogram.len(),
64        spectrogram.first().map_or(0, |v| v.len())
65    );
66
67    // Return the spectrogram
68    spectrogram
69}

71/// Downsample a spectrogram by averaging adjacent time frames into buckets.
72/// Returns the data unchanged if it already has fewer frames than `max_time_bins`.
73pub fn downsample_spectrogram(data: Vec<Vec<f32>>, max_time_bins: usize) -> Vec<Vec<f32>> {
74    let n = data.len();
75    if n <= max_time_bins || n == 0 {
76        return data;
77    }
78
79    let freq_bins = data[0].len();
80    let bucket_size = n.div_ceil(max_time_bins); // ceil division
81
82    info!(
83        "Downsampling spectrogram: {} frames -> ~{} frames (bucket_size={})",
84        n, max_time_bins, bucket_size
85    );
86
87    let mut result = Vec::with_capacity(max_time_bins);
88    for chunk in data.chunks(bucket_size) {
89        let mut avg = vec![0.0f32; freq_bins];
90        for frame in chunk {
91            for (i, &val) in frame.iter().enumerate() {
92                avg[i] += val;
93            }
94        }
95        let scale = 1.0 / chunk.len() as f32;
96        for v in &mut avg {
97            *v *= scale;
98        }
99        result.push(avg);
100    }
101
102    result
103}

109    #[test]
110    fn test_compute_spectrum() {
111        // Create a test signal with two frequencies
112        let sample_rate = 44100;
113        let duration = 0.5; // 500ms
114        let num_samples = (sample_rate as f32 * duration) as usize;
115
116        let mut samples = Vec::with_capacity(num_samples);
117        for i in 0..num_samples {
118            let t = i as f32 / sample_rate as f32;
119            // Mix 440Hz and 880Hz
120            samples.push(0.5 * (2.0 * PI * 440.0 * t).sin() + 0.3 * (2.0 * PI * 880.0 * t).sin());
121        }
122
123        let spectrogram = compute_spectrum(&samples, sample_rate);
124
125        // Basic verification
126        assert!(!spectrogram.is_empty());
127        assert!(!spectrogram[0].is_empty());
128    }
129
130    #[test]
131    fn test_downsample_no_op_when_small() {
132        let data = vec![vec![1.0, 2.0], vec![3.0, 4.0]];
133        let result = downsample_spectrogram(data.clone(), 10);
134        assert_eq!(result, data);
135    }

137    #[test]
138    fn test_downsample_averages_buckets() {
139        // 4 frames -> 2 buckets of size 2
140        let data = vec![
141            vec![2.0, 4.0],
142            vec![4.0, 6.0],
143            vec![10.0, 20.0],
144            vec![30.0, 40.0],
145        ];
146        let result = downsample_spectrogram(data, 2);
147        assert_eq!(result.len(), 2);
148        assert_eq!(result[0], vec![3.0, 5.0]); // avg of [2,4] and [4,6]
149        assert_eq!(result[1], vec![20.0, 30.0]); // avg of [10,20] and [30,40]
150    }
151
152    #[test]
