3/// Estimate pitch using autocorrelation
4pub fn estimate_pitch(samples: &[f32], sample_rate: u32) -> Option<f32> {
5    if samples.is_empty() {
6        return None;
7    }
8
9    // Use a section of the sample for pitch detection
10    let window_size = sample_rate as usize / 10; // 100ms window
11    let samples_to_use = if samples.len() > window_size {
12        &samples[0..window_size]
13    } else {
14        samples
15    };
16
17    // Compute autocorrelation
18    let mut autocorrelation = Vec::with_capacity(samples_to_use.len());
19    for lag in 0..samples_to_use.len() {
20        let mut sum = 0.0;
21        for i in 0..(samples_to_use.len() - lag) {
22            sum += samples_to_use[i] * samples_to_use[i + lag];
23        }
24        autocorrelation.push(sum);
25    }
26
27    // Find the highest peak after the first zero crossing
28    let mut zero_crossed = false;
29    let mut max_val = 0.0;
30    let mut max_lag = 0;
31
32    for (lag, &val) in autocorrelation.iter().enumerate().skip(1) {
33        if !zero_crossed && val <= 0.0 {
34            zero_crossed = true;
35        } else if zero_crossed && val > max_val {
36            max_val = val;
37            max_lag = lag;
38        }
39    }
40
41    // Calculate pitch
42    if max_lag > 0 {
43        let pitch = sample_rate as f32 / max_lag as f32;
44        info!("Estimated pitch: {:.2} Hz", pitch);
45        Some(pitch)
46    } else {
47        None
48    }
49}

51/// Converts frequency to musical note
52pub fn frequency_to_note(frequency: f32) -> String {
53    // A4 = 440Hz is our reference
54    const A4_FREQ: f32 = 440.0;
55    const A4_MIDI: i32 = 69; // MIDI note number for A4
56
57    // Note names with sharps
58    const NOTE_NAMES: [&str; 12] = [
59        "C", "C#", "D", "D#", "E", "F", "F#", "G", "G#", "A", "A#", "B",
60    ];
61
62    if frequency <= 0.0 {
63        return "?".to_string();
64    }
65
66    // Calculate MIDI note number (which can be fractional)
67    let midi_note = 12.0 * (frequency / A4_FREQ).log2() + A4_MIDI as f32;
68
69    // Get the closest MIDI note
70    let closest_midi = midi_note.round() as i32;
71
72    // Calculate cents deviation from the closest note
73    let cents = (midi_note - closest_midi as f32) * 100.0;
74
75    // Get the note name and octave
76    let note_index = ((closest_midi % 12) + 12) % 12; // Ensure positive
77    let octave = (closest_midi / 12) - 1; // MIDI octave system
78
79    // Format the note with octave and cents deviation
80    format!(
81        "{}{:}{:+.0}¢",
82        NOTE_NAMES[note_index as usize], octave, cents
83    )
84}

86#[cfg(test)]
87mod tests {
88    use super::*;
89    use std::f32::consts::PI;

91    #[test]
92    fn test_pitch_detection() {
93        // Create a sine wave at 440 Hz (A4)
94        let sample_rate = 44100;
95        let frequency = 440.0;
96        let duration = 0.2; // 200ms
97        let num_samples = (sample_rate as f32 * duration) as usize;
98
99        let mut samples = Vec::with_capacity(num_samples);
100        for i in 0..num_samples {
101            let t = i as f32 / sample_rate as f32;
102            samples.push((2.0 * PI * frequency * t).sin());
103        }
104
105        let detected_pitch = estimate_pitch(&samples, sample_rate);
106
107        assert!(detected_pitch.is_some());
108        let pitch = detected_pitch.unwrap();
109
110        // Allow 5% tolerance for pitch detection
111        assert!((pitch - 440.0).abs() < 22.0);
112    }
113
114    #[test]
115    fn test_frequency_to_note() {
116        // Test exact frequencies
117        assert_eq!(frequency_to_note(440.0), "A4+0¢");
118        assert_eq!(frequency_to_note(261.63), "C4+0¢");
119
120        // Test frequencies with cents deviation
121        let slightly_sharp_a = 440.0 * 2.0_f32.powf(0.1 / 12.0); // 10 cents sharp
122        assert!(frequency_to_note(slightly_sharp_a).contains("+10¢"));
123
124        let slightly_flat_a = 440.0 * 2.0_f32.powf(-0.15 / 12.0); // 15 cents flat
125        assert!(frequency_to_note(slightly_flat_a).contains("-15¢"));
126    }
