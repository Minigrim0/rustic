5/// Application state: holds the currently loaded audio file and its precomputed summary.
6/// Managed by Tauri via `RwLock<AudioState>`.
7pub struct AudioState {
8    pub buffer: Option<AudioBuffer>,
9    pub summary: Option<AudioSummary>,
10    pub loader: AudioLoader,
11}
12
13impl AudioState {
14    /// Converts a time range (seconds) to a sample index range clamped to the buffer bounds.
15    /// Returns `None` if no buffer is loaded or the range is empty.
16    pub fn time_to_sample_range(&self, start: f64, end: f64) -> Option<(usize, usize)> {
17        let buffer = self.buffer.as_ref()?;
18        let sr = buffer.sample_rate() as f64;
19        let total = buffer.samples().len();
20
21        let s = (start * sr).round().max(0.0) as usize;
22        let e = (end * sr).round().min(total as f64) as usize;
23
24        if s >= e {
25            return None;
26        }
27
28        Some((s, e))
29    }
