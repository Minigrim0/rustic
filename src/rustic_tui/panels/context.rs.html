1use ratatui::{

9/// Context information displayed in the reference panel.
10#[derive(Debug, Clone)]
11pub struct ContextInfo {
12    /// Available instruments and their status.
13    pub instruments: Vec<InstrumentInfo>,
14    /// Available keybinding hints for the current mode.
15    pub keybindings: Vec<(String, String)>,
16    /// Current playback / engine state summary.
17    pub engine_status: String,
18}
19
20#[derive(Debug, Clone)]
21pub struct InstrumentInfo {
22    pub name: String,
23    pub active: bool,
24    pub voice_count: usize,
25}
26
27impl Default for ContextInfo {
28    fn default() -> Self {
29        Self {
30            instruments: Vec::new(),
31            keybindings: Vec::new(),
32            engine_status: "Engine: idle".to_string(),
33        }
34    }
35}
36
37/// The context / reference panel (Column 3).
38/// Shows instrument state, keybinding hints, and engine status.
39pub struct ContextPanel<'a> {
40    info: &'a ContextInfo,
41    focused: bool,
42}
43
44impl<'a> ContextPanel<'a> {
45    pub fn new(info: &'a ContextInfo, focused: bool) -> Self {
46        Self { info, focused }
47    }
48}
49
50impl Widget for ContextPanel<'_> {
51    fn render(self, area: Rect, buf: &mut RBuf) {
52        let border_style = if self.focused {
53            Style::default().fg(Color::Cyan)
54        } else {
55            Style::default().fg(Color::DarkGray)
56        };
57
58        let block = Block::default()
59            .borders(Borders::ALL)
60            .border_style(border_style)
61            .title(" Context ");
62
63        let inner = block.inner(area);
64        block.render(area, buf);
65
66        if inner.width == 0 || inner.height == 0 {
67            return;
68        }
69
70        let mut lines: Vec<Line> = Vec::new();
71
72        // Engine status
73        lines.push(Line::from(vec![Span::styled(
74            "Engine Status",
75            Style::default()
76                .fg(Color::Cyan)
77                .add_modifier(Modifier::BOLD | Modifier::UNDERLINED),
78        )]));
79        lines.push(Line::from(vec![Span::styled(
80            format!("  {}", self.info.engine_status),
81            Style::default().fg(Color::White),
82        )]));
83        lines.push(Line::from(""));
84
85        // Instruments section
86        lines.push(Line::from(vec![Span::styled(
87            "Instruments",
88            Style::default()
89                .fg(Color::Cyan)
90                .add_modifier(Modifier::BOLD | Modifier::UNDERLINED),
91        )]));
92
93        if self.info.instruments.is_empty() {
94            lines.push(Line::from(vec![Span::styled(
95                "  (none loaded)",
96                Style::default().fg(Color::DarkGray),
97            )]));
98        } else {
99            for inst in &self.info.instruments {
100                let status_color = if inst.active {
101                    Color::Green
102                } else {
103                    Color::DarkGray
104                };
105                let indicator = if inst.active { "+" } else { "-" };
106                lines.push(Line::from(vec![
107                    Span::styled(
108                        format!("  {} ", indicator),
109                        Style::default().fg(status_color),
110                    ),
111                    Span::styled(inst.name.clone(), Style::default().fg(Color::White)),
112                    Span::styled(
113                        format!(" ({} voices)", inst.voice_count),
114                        Style::default().fg(Color::DarkGray),
115                    ),
116                ]));
117            }
118        }
119        lines.push(Line::from(""));
120
121        // Keybinding hints
122        lines.push(Line::from(vec![Span::styled(
123            "Keybindings",
124            Style::default()
125                .fg(Color::Cyan)
126                .add_modifier(Modifier::BOLD | Modifier::UNDERLINED),
127        )]));
128
129        for (key, desc) in &self.info.keybindings {
130            lines.push(Line::from(vec![
131                Span::styled(
132                    format!("  {:>12}", key),
133                    Style::default()
134                        .fg(Color::Yellow)
135                        .add_modifier(Modifier::BOLD),
136                ),
137                Span::styled(format!("  {}", desc), Style::default().fg(Color::White)),
138            ]));
139        }
140
141        let paragraph = Paragraph::new(lines).wrap(Wrap { trim: false });
142        paragraph.render(inner, buf);
143    }
144}
