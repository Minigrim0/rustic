6/// Extracts the range parameter from the token stream.
7fn extract_range_parameter(
8    field_name: String,
9    title: String,
10    stream: TokenStream,
11) -> Parameter<String> {
12    let mut values = stream.into_iter();
13    let minimum = values
14        .next()
15        .unwrap_or_else(|| panic!("No minimum value found"));
16    let maximum = values
17        .next()
18        .unwrap_or_else(|| panic!("No maximum value found"));
19    let default = values
20        .next()
21        .unwrap_or_else(|| panic!("No default value found"));
22
23    let minimum = if let TokenTree::Literal(minimum) = minimum {
24        minimum.to_string().parse().unwrap()
25    } else {
26        panic!("Minimum value is not a literal");
27    };
28
29    let maximum = if let TokenTree::Literal(maximum) = maximum {
30        maximum.to_string().parse().unwrap()
31    } else {
32        panic!("Maximum value is not a literal");
33    };
34
35    let default = if let TokenTree::Literal(default) = default {
36        default.to_string().parse().unwrap()
37    } else {
38        panic!("Default value is not a literal");
39    };
40
41    Parameter::Range {
42        title,
43        field_name,
44        min: minimum,
45        max: maximum,
46        default,
47        value: default,
48    }
49}

51/// Extracts the toggle parameter from the token stream.
52fn extract_toggle_parameter(
53    field_name: String,
54    title: String,
55    stream: TokenStream,
56) -> Parameter<String> {
57    let mut values = stream.into_iter();
58    if let Some(value) = values.next() {
59        let value = if let TokenTree::Literal(value) = value {
60            value.to_string().parse().unwrap()
61        } else {
62            panic!("Value is not a literal");
63        };
64
65        Parameter::Toggle {
66            title,
67            field_name,
68            value,
69            default: value,
70        }
71    } else {
72        panic!("No value found for toggle parameter");
73    }
74}

76/// Extracts the float parameter from the token stream.
77/// A float parameter is an unbounded parameter (contrary to
78/// a range parameter).
79fn extract_float_parameter(
80    field_name: String,
81    title: String,
82    stream: TokenStream,
83) -> Parameter<String> {
84    let value = stream.into_iter().next().unwrap();
85    let value = if let TokenTree::Literal(value) = value {
86        value.to_string().parse().unwrap()
87    } else {
88        panic!("Value is not a literal");
89    };
90
91    Parameter::Float {
92        title,
93        field_name,
94        default: value,
95        value,
96    }
97}

99/// Iterates through the token stream of a filter_parameter
100/// attribute to extract the parameter name and any additional
101/// information.
102pub fn extract_parameter(name: String, token_stream: TokenStream) -> Parameter<String> {
103    let mut stream = token_stream.into_iter();
104    let param_type = if let Some(TokenTree::Ident(token)) = stream.next() {
105        token.to_string()
106    } else {
107        panic!("No parameter type found");
108    };
109
110    let values = stream
111        .filter(|e| {
112            if let TokenTree::Punct(punct) = e {
113                punct.as_char() != ','
114            } else {
115                true
116            }
117        })
118        .collect();
119
120    let parameter_title: String = name.to_case(Case::Title);
121
122    match param_type.as_str() {
123        "range" => extract_range_parameter(name, parameter_title, values),
124        "toggle" => extract_toggle_parameter(name, parameter_title, values),
125        "float" => extract_float_parameter(name, parameter_title, values),
126        any => panic!("Unknown parameter type: {any}"),
127    }
128}
