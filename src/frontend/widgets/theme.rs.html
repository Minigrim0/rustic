13/// Theme options available in the application
14#[derive(Debug, Clone, Copy, PartialEq, Eq)]
15pub enum ThemeChoice {
16    /// Dark theme (default)
17    Dark,
18    /// Light theme
19    Light,
20    /// High contrast theme for accessibility
21    HighContrast,
22    /// Custom theme (user-defined)
23    Custom,
24}
25
26impl ThemeChoice {
27    /// Convert theme choice to string representation
28    pub fn as_string(&self) -> &'static str {
29        match self {
30            ThemeChoice::Dark => "Dark",
31            ThemeChoice::Light => "Light",
32            ThemeChoice::HighContrast => "High Contrast",
33            ThemeChoice::Custom => "Custom",
34        }
35    }
36
37    /// Get all available theme choices
38    pub fn all() -> Vec<ThemeChoice> {
39        vec![
40            ThemeChoice::Dark,
41            ThemeChoice::Light,
42            ThemeChoice::HighContrast,
43            ThemeChoice::Custom,
44        ]
45    }
46}

48/// Configure the application theme based on the selected theme choice
49///
50/// # Arguments
51///
52/// * `theme` - The selected theme choice
53/// * `ctx` - The egui context to apply the theme to
54///
55/// # Examples
56///
57/// ```
58/// use frontend::widgets::{configure_theme, ThemeChoice};
59///
60/// fn update_theme(ctx: &egui::Context) {
61///     configure_theme(ThemeChoice::Dark, ctx);
62/// }
63/// ```
64pub fn configure_theme(theme: ThemeChoice, ctx: &egui::Context) {
65    let style = match theme {
66        ThemeChoice::Dark => create_dark_theme(),
67        ThemeChoice::Light => create_light_theme(),
68        ThemeChoice::HighContrast => create_high_contrast_theme(),
69        ThemeChoice::Custom => create_custom_theme(),
70    };
71
72    ctx.set_style(style);
73}

75/// Create the dark theme style
76fn create_dark_theme() -> Style {
77    let mut style = Style::default();
78    style.spacing.item_spacing = egui::vec2(spacing::ITEM_SPACING, spacing::ITEM_SPACING);
79    style.spacing.window_margin = margins::DEFAULT;
80    style.spacing.button_padding = egui::vec2(spacing::CONTAINER_PADDING, spacing::SMALL_PADDING);
81
82    let mut visuals = Visuals::dark();
83    visuals.override_text_color = Some(colors::TEXT);
84    visuals.widgets.noninteractive.bg_fill = colors::BACKGROUND_DARK;
85    visuals.widgets.inactive.bg_fill = colors::BACKGROUND_MED;
86    visuals.widgets.hovered.bg_fill = colors::BACKGROUND_LIGHT;
87    visuals.widgets.active.bg_fill = colors::PRIMARY;
88    visuals.widgets.active.fg_stroke = Stroke::new(1.0, Color32::WHITE);
89    visuals.widgets.open.bg_fill = colors::BACKGROUND_LIGHT;
90
91    visuals.selection = Selection {
92        bg_fill: colors::PRIMARY.linear_multiply(0.8),
93        stroke: strokes::HIGHLIGHT,
94    };
95
96    visuals.window_rounding = Rounding::same(rounding::DEFAULT);
97    visuals.window_shadow.extrusion = 8.0;
98
99    visuals.menu_rounding = Rounding::same(rounding::DEFAULT);
100    // Note: panel_rounding is not available in newer egui versions
101    // Individual widgets handle their own rounding
102
103    visuals.popup_shadow.extrusion = 6.0;
104
105    style.visuals = visuals;
106    style
107}

109/// Create the light theme style
110fn create_light_theme() -> Style {
111    let mut style = Style::default();
112    style.spacing.item_spacing = egui::vec2(spacing::ITEM_SPACING, spacing::ITEM_SPACING);
113    style.spacing.window_margin = margins::DEFAULT;
114    style.spacing.button_padding = egui::vec2(spacing::CONTAINER_PADDING, spacing::SMALL_PADDING);
115
116    let mut visuals = Visuals::light();
117    visuals.override_text_color = Some(Color32::from_rgb(20, 20, 20));
118
119    // Define light theme specific colors
120    let bg_fill = Color32::from_rgb(240, 240, 245);
121    let bg_med = Color32::from_rgb(220, 220, 230);
122    let bg_light = Color32::from_rgb(230, 230, 240);
123    let primary = Color32::from_rgb(40, 120, 200);
124
125    visuals.widgets.noninteractive.bg_fill = bg_fill;
126    visuals.widgets.inactive.bg_fill = bg_med;
127    visuals.widgets.hovered.bg_fill = bg_light;
128    visuals.widgets.active.bg_fill = primary;
129    visuals.widgets.active.fg_stroke = Stroke::new(1.0, Color32::WHITE);
130    visuals.widgets.open.bg_fill = bg_light;
131
132    visuals.selection = Selection {
133        bg_fill: primary.linear_multiply(0.8),
134        stroke: Stroke::new(2.0, primary),
135    };
136
137    visuals.window_rounding = Rounding::same(rounding::DEFAULT);
138    visuals.window_shadow.extrusion = 8.0;
139
140    visuals.menu_rounding = Rounding::same(rounding::DEFAULT);
141    // Panel rounding handled through individual frame styling
142
143    visuals.popup_shadow.extrusion = 6.0;
144
145    style.visuals = visuals;
146    style

149/// Create a high contrast theme for accessibility
150fn create_high_contrast_theme() -> Style {
151    let mut style = Style::default();
152    style.spacing.item_spacing = egui::vec2(spacing::ITEM_SPACING, spacing::ITEM_SPACING);
153    style.spacing.window_margin = margins::LARGE;
154    style.spacing.button_padding = egui::vec2(spacing::LARGE_PADDING, spacing::CONTAINER_PADDING);
155
156    let mut visuals = Visuals::dark();
157
158    // Define high contrast specific colors
159    let bg_fill = Color32::BLACK;
160    let text_color = Color32::WHITE;
161    let primary = Color32::from_rgb(255, 255, 0); // Yellow for high contrast
162    let secondary = Color32::from_rgb(50, 255, 255); // Cyan for high contrast
163
164    visuals.override_text_color = Some(text_color);
165
166    visuals.widgets = Widgets::default();
167    visuals.widgets.noninteractive.bg_fill = bg_fill;
168    visuals.widgets.noninteractive.fg_stroke = Stroke::new(1.5, text_color);
169
170    visuals.widgets.inactive.bg_fill = bg_fill;
171    visuals.widgets.inactive.fg_stroke = Stroke::new(1.5, text_color);
172
173    visuals.widgets.hovered.bg_fill = primary;
174    visuals.widgets.hovered.fg_stroke = Stroke::new(2.0, Color32::BLACK);
175
176    visuals.widgets.active.bg_fill = secondary;
177    visuals.widgets.active.fg_stroke = Stroke::new(2.0, Color32::BLACK);
178
179    visuals.widgets.open.bg_fill = primary.linear_multiply(0.8);
180    visuals.widgets.open.fg_stroke = Stroke::new(2.0, Color32::BLACK);
181
182    visuals.selection = Selection {
183        bg_fill: primary,
184        stroke: Stroke::new(2.0, text_color),
185    };
186
187    visuals.window_rounding = Rounding::same(rounding::SLIGHT);
188    visuals.window_shadow.extrusion = 10.0;
189
190    visuals.menu_rounding = Rounding::same(rounding::SLIGHT);
191    // Panel rounding handled through individual frame styling
192
193    visuals.popup_shadow.extrusion = 8.0;
194
195    style.visuals = visuals;
196    style
197}

217/// Apply a specific scale factor to the UI
218///
219/// # Arguments
220///
221/// * `scale` - The scaling factor (1.0 is the default scale)
222/// * `ctx` - The egui context to apply the scaling to
223///
224/// # Examples
225///
226/// ```
227/// use frontend::widgets::apply_scaling;
228///
229/// fn update_scale(ctx: &egui::Context) {
230///     apply_scaling(1.2, ctx); // Scale UI by 120%
231/// }
232/// ```
233pub fn apply_scaling(scale: f32, ctx: &egui::Context) {
234    ctx.set_pixels_per_point(scale);
235}
