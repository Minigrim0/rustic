20/// Live Playing tab for real-time instrument performance
21pub struct LivePlayingTab {
22    pub is_enabled: bool,
23    channels: Vec<ChannelData>,
24    _active_notes: Vec<(usize, u8)>, // (channel_index, note_value)
25}

27impl LivePlayingTab {
28    /// Create a new LivePlayingTab with placeholder data
29    pub fn new() -> Self {
30        // Create placeholder channel data
31        let mut channels = Vec::new();
32
33        // Instrument names for placeholder data
34        let instrument_names = [
35            "Grand Piano",
36            "Electric Piano",
37            "Organ",
38            "Synth Lead",
39            "Synth Pad",
40            "Strings",
41            "Guitar",
42            "Bass",
43            "Drums",
44            "Brass",
45            "Flute",
46            "Clarinet",
47            "Saxophone",
48            "Violin",
49            "Cello",
50            "Harp",
51            "Marimba",
52            "Vibraphone",
53            "Accordion",
54            "Voice",
55        ];
56
57        // Generate 20 channels with placeholder data
58        for i in 0..20 {
59            let instrument_idx = i % instrument_names.len();
60            channels.push(ChannelData {
61                number: i + 1,
62                instrument_name: instrument_names[instrument_idx].to_string(),
63                octave: 4 + (i % 5) as u8,
64                volume: 0.75 + (i as f32 * 0.01),
65                is_muted: i % 7 == 0,
66                is_linked: i % 2 == 0 && i > 0,
67                is_playing: false,
68            });
69        }
70
71        LivePlayingTab {
72            is_enabled: true,
73            channels,
74            _active_notes: Vec::new(),
75        }
76    }
77
78    /// Simulate note activation for visual feedback
79    fn _activate_note(&mut self, channel_idx: usize, note_value: u8) {
80        if channel_idx < self.channels.len() {
81            self.channels[channel_idx].is_playing = true;
82            self._active_notes.push((channel_idx, note_value));
83        }
84    }
85
86    /// Simulate note deactivation
87    fn _deactivate_note(&mut self, channel_idx: usize, note_value: u8) {
88        if channel_idx < self.channels.len() {
89            // Remove the note from active notes
90            self._active_notes
91                .retain(|&(ch, note)| ch != channel_idx || note != note_value);
92
93            // If no more notes are active for this channel, set is_playing to false
94            if !self._active_notes.iter().any(|&(ch, _)| ch == channel_idx) {
95                self.channels[channel_idx].is_playing = false;
96            }
97        }
98    }

100    /// Draw the channel grid
101    fn draw_channel_grid(&mut self, ui: &mut Ui) {
102        // Calculate number of columns based on available width
103        let available_width = ui.available_width();
104        let column_width = 380.0; // Width of a single channel column
105        let columns_per_row = (available_width / column_width).max(1.0).floor() as usize;
106
107        // Group channels into rows
108        for chunk_index in 0..self.channels.len().div_ceil(columns_per_row) {
109            ui.horizontal(|ui| {
110                let start_idx = chunk_index * columns_per_row;
111                let end_idx = (start_idx + columns_per_row).min(self.channels.len());
112
113                for channel_idx in start_idx..end_idx {
114                    let channel = &self.channels[channel_idx];
115
116                    // Channel card using SectionContainer
117                    SectionContainer::new(&format!(
118                        "#{}: {}",
119                        channel.number, channel.instrument_name
120                    ))
121                    .show_title(false)
122                    .with_frame(true)
123                    .show(ui, |ui| {
124                        ui.set_min_width(column_width);
125                        ui.set_max_width(column_width);
126
127                        // Channel header
128                        ui.horizontal(|ui| {
129                            // Channel number and name with visual playing indicator
130                            let bg_color = if channel.is_playing {
131                                Color32::from_rgb(70, 120, 70) // Green background when playing
132                            } else if channel.is_muted {
133                                Color32::from_rgb(120, 70, 70) // Red background when muted
134                            } else {
135                                Color32::from_rgb(60, 60, 80) // Default dark blue-gray
136                            };
137
138                            ui.painter()
139                                .rect_filled(ui.min_rect().expand(2.0), 4.0, bg_color);
140
141                            let channel_text =
142                                format!("#{}: {}", channel.number, channel.instrument_name);
143                            ui.label(RichText::new(channel_text).size(16.0).strong());
144
145                            // Right-aligned mute indicator
146                            ui.with_layout(
147                                egui::Layout::right_to_left(egui::Align::Center),
148                                |ui| {
149                                    let mute_text = if channel.is_muted {
150                                        "ðŸ”‡ Muted"
151                                    } else {
152                                        "ðŸ”Š"
153                                    };
154                                    ui.label(RichText::new(mute_text).color(if channel.is_muted {
155                                        Color32::LIGHT_RED
156                                    } else {
157                                        Color32::LIGHT_GREEN
158                                    }));
159                                },
160                            );
161                        });
162
163                        ui.add_space(4.0);
164
165                        // Channel details in a grid using DataGrid
166                        let octave_str = format!("{}", channel.octave);
167                        let volume_str = format!("{:.2}", channel.volume);
168                        let grid_data = vec![
169                            vec!["Octave:", octave_str.as_str()],
170                            vec!["Volume:", volume_str.as_str()],
171                            vec!["Linked:", if channel.is_linked { "Yes" } else { "No" }],
172                        ];
173
174                        // Apply custom colors for the "Linked" status
175                        let mut data_grid = DataGrid::new(channel.instrument_name.clone())
176                            .with_data(grid_data)
177                            .with_col_spacing(8.0)
178                            .with_row_spacing(4.0);
179
180                        if channel.is_linked {
181                            data_grid = data_grid.with_cell_color(2, 1, Color32::LIGHT_BLUE);
182                        } else {
183                            data_grid = data_grid.with_cell_color(2, 1, Color32::GRAY);
184                        }
185
186                        data_grid.show(ui);
187
188                        // Visual indicator for active playing
189                        if channel.is_playing {
190                            ui.add_space(4.0);
191
192                            // Activity indicator
193                            let (rect, _) = ui.allocate_exact_size(
194                                Vec2::new(ui.available_width(), 8.0),
195                                egui::Sense::hover(),
196                            );
197                            ui.painter().rect_filled(
198                                rect,
199                                4.0,
200                                Color32::from_rgb(100, 255, 100), // Bright green for activity
201                            );
202                        }
203                    });
204
205                    ui.add_space(8.0); // Space between channels
206                }
207            });
208
209            ui.add_space(8.0); // Space between rows
210        }
211    }

227                        log::info!("Live mode disabled");

281        // Only process keyboard events if the live mode is enabled
282        if self.is_enabled {
283            // TODO: Implement proper keyboard event handling for note playing
284            // Simulate some activity for demonstration purposes
285            // In a real implementation, this would be connected to actual note events

293                channel.is_playing = false;
294            }

296            // Activate a channel based on time

298                self.channels[active_time].is_playing = true;
299            }
300        }
301
302        // Scrollable area for the channel grid

305        });
