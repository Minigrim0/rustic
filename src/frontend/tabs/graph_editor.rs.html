27/// A simple node in the graph
28struct Node {
29    name: String,
30    node_type: NodeType,
31    position: Vec2,
32    params: Vec<(String, f32)>,
33}

35/// Types of nodes in the graph
36#[derive(Clone, Copy, PartialEq)]
37enum NodeType {
38    Generator,
39    Filter,
40    Sink,
41}
42
43impl NodeType {
44    fn color(&self) -> Color32 {
45        match self {
46            NodeType::Generator => Color32::from_rgb(100, 150, 200), // Blue
47            NodeType::Filter => Color32::from_rgb(100, 200, 100),    // Green
48            NodeType::Sink => Color32::from_rgb(200, 100, 100),      // Red
49        }
50    }
51
52    fn name(&self) -> &'static str {
53        match self {
54            NodeType::Generator => "Generator",
55            NodeType::Filter => "Filter",
56            NodeType::Sink => "Sink",
57        }
58    }
59}

204    /// Draw a node on the canvas
205    fn draw_node(&self, ui: &mut Ui, node: &Node, is_selected: bool) {
206        let node_color = node.node_type.color();
207        let node_size = Vec2::new(180.0, 120.0);
208        let node_pos = node.position + self.canvas_offset;
209
210        // Node background
211        let node_rect = egui::Rect::from_min_size(egui::pos2(node_pos.x, node_pos.y), node_size);
212
213        // Draw node background
214        ui.painter().rect(
215            node_rect,
216            4.0,
217            node_color,
218            if is_selected {
219                Stroke::new(2.0, Color32::WHITE)
220            } else {
221                Stroke::new(1.0, Color32::BLACK)
222            },
223        );
224
225        // Draw node title bar
226        let title_rect = egui::Rect::from_min_size(node_rect.min, Vec2::new(node_size.x, 24.0));
227
228        ui.painter().rect(
229            title_rect,
230            4.0,
231            node_color.linear_multiply(0.8),
232            Stroke::NONE,
233        );
234
235        // Draw node title
236        ui.painter().text(
237            title_rect.center(),
238            egui::Align2::CENTER_CENTER,
239            &node.name,
240            egui::FontId::proportional(14.0),
241            Color32::WHITE,
242        );
243
244        // Draw node type
245        ui.painter().text(
246            egui::pos2(title_rect.min.x + 5.0, title_rect.min.y + 5.0),
247            egui::Align2::LEFT_TOP,
248            node.node_type.name(),
249            egui::FontId::proportional(10.0),
250            Color32::from_rgba_premultiplied(255, 255, 255, 180),
251        );
252
253        // Draw parameters
254        for (i, (name, value)) in node.params.iter().enumerate() {
255            let param_y = node_rect.min.y + 30.0 + i as f32 * 20.0;
256
257            // Parameter name
258            ui.painter().text(
259                egui::pos2(node_rect.min.x + 10.0, param_y),
260                egui::Align2::LEFT_CENTER,
261                name,
262                egui::FontId::proportional(12.0),
263                Color32::WHITE,
264            );
265
266            // Parameter value
267            ui.painter().text(
268                egui::pos2(node_rect.max.x - 10.0, param_y),
269                egui::Align2::RIGHT_CENTER,
270                format!("{:.1}", value),
271                egui::FontId::proportional(12.0),
272                Color32::LIGHT_YELLOW,
273            );
274        }
275
276        // TODO: Implement proper drag-and-drop functionality for node parameters
277
278        // Draw connection points based on node type
279        if node.node_type != NodeType::Sink {
280            // Output port (on right side)
281            let out_pos = egui::pos2(node_rect.right(), node_rect.min.y + 60.0);
282            ui.painter().circle(
283                out_pos,
284                6.0,
285                Color32::LIGHT_GREEN,
286                Stroke::new(1.0, Color32::BLACK),
287            );
288        }
289
290        if node.node_type != NodeType::Generator {
291            // Input port (on left side)
292            let in_pos = egui::pos2(node_rect.left(), node_rect.min.y + 60.0);
293            ui.painter().circle(
294                in_pos,
295                6.0,
296                Color32::LIGHT_BLUE,
297                Stroke::new(1.0, Color32::BLACK),
298            );
299        }
300    }

302    /// Draw the canvas with nodes and connections
303    fn draw_canvas(&self, ui: &mut Ui) {
304        let canvas_rect = ui.available_rect_before_wrap();
305        ui.allocate_rect(canvas_rect, egui::Sense::click_and_drag());
306
307        // Draw a grid in the background
308        self.draw_grid(ui, canvas_rect);
309
310        // Draw connections
311        for (from_idx, to_idx) in &self.connections {
312            if *from_idx < self.nodes.len() && *to_idx < self.nodes.len() {
313                let from_node = &self.nodes[*from_idx];
314                let to_node = &self.nodes[*to_idx];
315
316                // Calculate connection points
317                let start_pos = egui::pos2(
318                    from_node.position.x + 180.0 + self.canvas_offset.x,
319                    from_node.position.y + 60.0 + self.canvas_offset.y,
320                );
321
322                let end_pos = egui::pos2(
323                    to_node.position.x + self.canvas_offset.x,
324                    to_node.position.y + 60.0 + self.canvas_offset.y,
325                );
326
327                // Draw a bezier curve for the connection
328                self.draw_connection(ui, start_pos, end_pos);
329            }
330        }
331
332        // Draw nodes
333        for (idx, node) in self.nodes.iter().enumerate() {
334            self.draw_node(ui, node, Some(idx) == self.selected_node);
335        }
336
337        // Show help window if enabled
338        if self.show_help {
339            self.draw_help_window(ui);
340        }
341    }

343    /// Draw a connection between two points
344    fn draw_connection(&self, ui: &mut Ui, start: egui::Pos2, end: egui::Pos2) {
345        let control_distance = (end.x - start.x).abs().max(50.0) * 0.5;
346        let control1 = egui::pos2(start.x + control_distance, start.y);
347        let control2 = egui::pos2(end.x - control_distance, end.y);
348
349        ui.painter().add(egui::Shape::CubicBezier(
350            egui::epaint::CubicBezierShape::from_points_stroke(
351                [start, control1, control2, end],
352                false,
353                Color32::TRANSPARENT,
354                Stroke::new(2.0, Color32::LIGHT_BLUE),
355            ),
356        ));
357    }

359    /// Draw a grid background
360    fn draw_grid(&self, ui: &mut Ui, rect: egui::Rect) {
361        let grid_color = Color32::from_rgb(50, 50, 50);
362        let grid_size = 20.0;
363
364        // Draw vertical grid lines
365        for x in (rect.min.x as i32..(rect.max.x as i32)).step_by(grid_size as usize) {
366            ui.painter().line_segment(
367                [
368                    egui::pos2(x as f32, rect.min.y),
369                    egui::pos2(x as f32, rect.max.y),
370                ],
371                Stroke::new(0.5, grid_color),
372            );
373        }
374
375        // Draw horizontal grid lines
376        for y in (rect.min.y as i32..(rect.max.y as i32)).step_by(grid_size as usize) {
377            ui.painter().line_segment(
378                [
379                    egui::pos2(rect.min.x, y as f32),
380                    egui::pos2(rect.max.x, y as f32),
381                ],
382                Stroke::new(0.5, grid_color),
383            );
384        }
385    }
