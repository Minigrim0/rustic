1//! Simple note representation for audio processing
2//!
3//! This module provides a lightweight note type that combines a musical note
4//! name with an octave number, suitable for audio processing and instrument control.
5
6use super::tones::NOTES;
7use serde::{Deserialize, Serialize};
8
9/// A simple musical note with its octave
10///
11/// This is a lightweight representation of a musical note consisting of:
12/// - A note name from the chromatic scale (C, C#, D, etc.)
13/// - An octave number (typically 0-8)
14///
15/// # Examples
16///
17/// ```
18/// use rustic::core::utils::{Note, NOTES};
19///
20/// let middle_c = Note(NOTES::C, 4);
21/// let a440 = Note(NOTES::A, 4);
22/// ```
23#[derive(Hash, Eq, PartialEq, Debug, Copy, Clone, Serialize, Deserialize)]
24pub struct Note(pub NOTES, pub u8);
25
26impl Note {
27    /// Create a new note from a note name and octave
28    pub fn new(note: NOTES, octave: u8) -> Self {
29        Self(note, octave)
30    }
31
32    /// Get the note name
33    pub fn note(&self) -> NOTES {
34        self.0
35    }
36
37    /// Get the octave
38    pub fn octave(&self) -> u8 {
39        self.1
40    }
41
42    /// Calculate the frequency of this note in Hz
43    ///
44    /// Uses equal temperament tuning with A4 = 440 Hz
45    pub fn frequency(&self) -> f32 {
46        use super::tones::TONES_FREQ;
47
48        let octave = self.1 as usize;
49        let note_index = self.0 as usize;
50
51        if octave < TONES_FREQ[0].len() {
52            TONES_FREQ[note_index][octave]
53        } else {
54            // For octaves beyond our table, use the formula
55            let base_freq = TONES_FREQ[note_index][0];
56            base_freq * (2.0_f32).powi(octave as i32)
57        }
58    }
59
60    /// Convert from MIDI note number
61    ///
62    /// MIDI note 60 = C4 (middle C)
63    pub fn from_midi(midi_note: u8) -> Self {
64        let octave = midi_note / 12;
65        let note_index = midi_note % 12;
66        let note = NOTES::from(note_index);
67        Self(note, octave.saturating_sub(1)) // MIDI octave -1 adjustment
68    }
69
70    /// Convert to MIDI note number
71    ///
72    /// Returns the MIDI note number (0-127) for this note
73    pub fn to_midi(&self) -> u8 {
74        let octave_offset = (self.1 + 1) * 12; // MIDI octave +1 adjustment
75        let note_offset = self.0 as u8;
76        (octave_offset + note_offset).min(127)
77    }
78
79    /// Transpose the note by a number of semitones
80    ///
81    /// Positive values transpose up, negative values transpose down
82    pub fn transpose(self, semitones: i8) -> Self {
83        let midi = self.to_midi() as i8;
84        let new_midi = (midi + semitones).clamp(0, 127) as u8;
85        Self::from_midi(new_midi)
86    }
87}
88
89impl std::fmt::Display for Note {
90    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
91        let note_name = match self.0 {
92            NOTES::C => "C",
93            NOTES::CS => "C#",
94            NOTES::D => "D",
95            NOTES::DS => "D#",
96            NOTES::E => "E",
97            NOTES::F => "F",
98            NOTES::FS => "F#",
99            NOTES::G => "G",
100            NOTES::GS => "G#",
101            NOTES::A => "A",
102            NOTES::AS => "A#",
103            NOTES::B => "B",
104        };
105        write!(f, "{}{}", note_name, self.1)
106    }
107}
