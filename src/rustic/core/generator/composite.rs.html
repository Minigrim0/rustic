9#[derive(Default, Debug, Serialize, Deserialize)]
10/// A generator that produces multiple tones. Each
11/// tone can have its own frequency relation, waveform,
12/// and envelopes.
13pub struct MultiToneGenerator {
14    base_frequency: f32,
15    tone_generators: Vec<SingleToneGenerator>,
16    mix_mode: super::prelude::MixMode,
17    global_pitch_envelope: Option<Box<dyn Envelope>>,
18    global_amplitude_envelope: Option<Box<dyn Envelope>>,
19    time: f32,
20    note_off: Option<f32>,
21}
22
23impl MultiToneGenerator {
24    pub fn new(
25        base_frequency: f32,
26        mut tone_generators: Vec<super::tone::SingleToneGenerator>,
27        mix_mode: super::prelude::MixMode,
28        global_pitch_envelope: Option<Box<dyn Envelope>>,
29        global_amplitude_envelope: Option<Box<dyn Envelope>>,
30    ) -> Self {
31        // Setup individual tone generators' frequencies
32        tone_generators.iter_mut().for_each(|tg| {
33            if !tg.has_frequency_relation() {
34                log::warn!("Adding a tone generator without a frequency relation to a composite generator. The generator will not get updated");
35            } else {
36                tg.update_frequency(base_frequency);
37            }
38        });
39
40        Self {
41            base_frequency,
42            tone_generators,
43            mix_mode,
44            global_pitch_envelope,
45            global_amplitude_envelope,
46            time: 0.0,
47            note_off: None,
48        }
49    }
50
51    pub fn start(&mut self) {
52        trace!("Composite Generator starting ({}Hz)", self.base_frequency);
53        self.time = 0.0;
54        self.note_off = None;
55        // Reset all child tone generators to ensure clean retriggering
56        self.tone_generators.iter_mut().for_each(|tg| tg.start());
57    }
58
59    pub fn stop(&mut self) {
60        trace!(
61            "Composite Generator stopping: {} ({}Hz)",
62            self.time, self.base_frequency
63        );
64        self.note_off = Some(self.time);
65        self.tone_generators.iter_mut().for_each(|tg| tg.stop());
66    }
67
68    pub fn completed(&self) -> bool {
69        // A composite generator is completed when all its tone generators are completed
70        // or when there's a note_off and sufficient time has passed for all envelopes to finish
71        if self.tone_generators.is_empty() {
72            return true;
73        }
74
75        // Check if all tone generators have completed
76        self.tone_generators.iter().all(|tg| tg.completed())
77    }
78
79    pub fn tick(&mut self, time_elapsed: f32) -> f32 {
80        let actual_elapsed = if let Some(envelope) = &self.global_pitch_envelope {
81            time_elapsed * envelope.at(self.time, self.note_off.unwrap_or(0.0))
82        } else {
83            time_elapsed
84        };
85
86        // Map true time elapsed for pitch bend
87        self.time += time_elapsed;
88
89        let values = self
90            .tone_generators
91            .iter_mut()
92            .map(|g| g.tick(actual_elapsed))
93            .collect::<Vec<f32>>();
94
95        let ampl = match self.mix_mode {
96            MixMode::Average => values.iter().sum::<f32>() / values.len() as f32,
97            MixMode::Multiply => values.iter().fold(1.0, |a, v| a * v),
98            MixMode::Max => values.iter().fold(f32::NEG_INFINITY, |a, v| a.max(*v)),
99            MixMode::Sum => values.iter().sum(),
100        };
101
102        if let Some(envelope) = &self.global_amplitude_envelope {
103            trace!(
104                "Composite Generator ticking: t:{} e:{} a:{} ae:{} ({}Hz)",
105                self.time,
106                envelope.at(self.time, self.note_off.unwrap_or(0.0)),
107                ampl,
108                ampl * envelope.at(self.time, self.note_off.unwrap_or(0.0)),
109                self.base_frequency
110            );
111
112            ampl * envelope.at(self.time, self.note_off.unwrap_or(0.0))
113        } else {
114            trace!(
115                "Composite Generator ticking: t:{} a:{} ({}Hz)",
116                self.time, ampl, self.base_frequency
117            );
118
119            ampl
120        }
121    }
122
123    pub fn add_tone(&mut self, tone: super::tone::SingleToneGenerator) {
124        self.tone_generators.push(tone);
125    }
126
127    pub fn set_base_frequency(&mut self, frequency: f32) {
128        self.base_frequency = frequency;
129        for generator in self.tone_generators.iter_mut() {
130            generator.update_frequency(frequency);
131        }
132    }
133
134    pub fn tone_count(&self) -> usize {
135        self.tone_generators.len()
136    }
137}
