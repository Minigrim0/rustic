18/// A Pipe & Filter system
19/// The system is composed of filters, sources and sinks.
20/// It is represented as a directed graph where the filters are the nodes
21/// The Sources & Sinks are special nodes that have respectively only outgoing (source) or incoming (sink) edges
22/// The edges represent the pipes between the filters
23/// Some filters have special output properties. E.g. the delay filter's input pipe is ignored (postponed) when
24/// the topology sorting is done, in order to avoid cycles. A system with cycles must include a delay or similar filter
25/// to break the cycle.
26///
27/// ```rust
28/// use rustic::core::graph::System;
29/// use rustic::core::filters::prelude::Tremolo;
30///
31/// // A simple system with one input and one output
32/// let mut system = System::<1, 1>::new();
33///
34/// // Adding a filter to the system
35/// let filter = Tremolo::new(20.0, 0.5, 1.5);
36/// let filter_index = system.add_filter(Box::from(filter));
37/// ```
38#[derive(Debug)]
39#[allow(clippy::type_complexity)]
40pub struct System<const INPUTS: usize, const OUTPUTS: usize> {
41    // The actual filter graph, from which the execution order is derived
42    // Each weight represents the port into which the filter is connected
43    graph: Graph<Box<dyn Filter>, (usize, usize)>,
44    // Each layer represent filters that can be run concurrently.
45    layers: Vec<Vec<usize>>,
46    // The sources of the system and the filters they are connected to
47    // The node index is the index of the filter that the source is connected to
48    // The second usize is the port of the filter that the source is connected to
49    sources: [(Box<dyn Source>, (NodeIndex<u32>, usize)); INPUTS],
50    // The sinks of the system.
51    // The node index is the index of the filter that the sink is connected to
52    // The second usize is the port of the filter that the sink is connected to
53    sinks: [((NodeIndex<u32>, usize), Box<dyn Sink>); OUTPUTS],
54}
55
56impl<const INPUTS: usize, const OUTPUTS: usize> Default for System<INPUTS, OUTPUTS> {
57    fn default() -> Self {
58        Self::new()
59    }

65    pub fn new() -> Self {
66        let sources: [(Box<dyn Source>, (NodeIndex<u32>, usize)); INPUTS] =
67            core::array::from_fn(|_| {

69                    simple_source(

74                                    .build(),

77                    ),
78                    (NodeIndex::new(0), 0),
79                )
80            });
81        let sinks: [((NodeIndex<u32>, usize), Box<dyn Sink>); OUTPUTS] =
82            core::array::from_fn(|_| ((NodeIndex::new(0), 0), simple_sink()));

84        System {
85            graph: Graph::new(),
86            layers: Vec::new(),
87            sources,
88            sinks,
89        }
90    }

92    /// Merges the two systems together to create a new one. The graphs are merged following the given mapping from sinks to sources.
93    /// Sinks to sources links are replaced with a simple combinator filter. The amount of input in the second system
94    /// should match the amount of output in the first system.
95    #[allow(clippy::type_complexity)]
96    pub fn merge<const T: usize>(
97        mut self,
98        other: System<OUTPUTS, T>,
99        mapping: Vec<(usize, usize)>,
100    ) -> System<INPUTS, T> {
101        // Contains the mapping other graph -> new graph
102        let mut new_edge_map: HashMap<NodeIndex, NodeIndex> = HashMap::new();
103
104        for (from, to) in mapping.iter() {
105            let (graph_b_source_descendant_index, graph_b_node_port) = other.sources[*to].1;
106            let source_descendant =
107                dyn_clone::clone_box(&*other.graph[graph_b_source_descendant_index]);
108
109            // Save the new index of the source descendant
110            let new_index = if let std::collections::hash_map::Entry::Vacant(e) =
111                new_edge_map.entry(graph_b_source_descendant_index)
112            {
113                let new_index = self.graph.add_node(source_descendant);
114                self.graph[new_index].set_index(new_index.index());
115                info!(
116                    "idx {} -> idx {}",
117                    graph_b_source_descendant_index.index(),
118                    new_index.index()
119                );
120                e.insert(new_index);
121                new_index
122            } else {
123                match new_edge_map.get(&graph_b_source_descendant_index) {
124                    Some(v) => *v,
125                    None => panic!("What ?"),
126                }
127            };
128
129            // Connect the sink's predecessors to the source's successors
130            let (sink_predecessor_id, sink_predecessor_port) = self.sinks[*from].0;
131            info!(
132                "Node {} -> Sink {} & Source {} -> Node {} => Node {} -> Node {}",
133                sink_predecessor_id.index(),
134                from,
135                to,
136                graph_b_source_descendant_index.index(),
137                sink_predecessor_id.index(),
138                new_index.index()
139            );
140            self.graph.add_edge(
141                sink_predecessor_id,
142                new_index,
143                (sink_predecessor_port, graph_b_node_port),
144            );
145        }
146
147        // Go through all nodes in the other graph and add them to the new graph
148        for node_index in other.graph.node_indices() {
149            // Skip already added nodes
150            if new_edge_map.contains_key(&node_index) {
151                info!("Node {} already pushed to new graph", node_index.index());
152                continue;
153            }
154
155            let node = dyn_clone::clone_box(&*other.graph[node_index]);
156            let new_index = self.graph.add_node(node);
157            new_edge_map.insert(node_index, new_index);
158        }
159
160        // Now that all nodes are added, connect the edges
161        for edge in other.graph.edge_indices() {
162            let (other_from, other_to) = other.graph.edge_endpoints(edge).unwrap();
163            let (from, to) = (new_edge_map[&other_from], new_edge_map[&other_to]);
164            info!(
165                "Edge ({}, {}) -> ({}, {})",
166                other_from.index(),
167                other_to.index(),
168                from.index(),
169                to.index()
170            );
171            let weight = other.graph[edge];
172            self.graph.add_edge(from, to, weight);
173        }
174
175        let new_sinks: [((NodeIndex<u32>, usize), Box<dyn Sink>); T] =
176            core::array::from_fn(|index| {
177                (
178                    (
179                        new_edge_map[&other.sinks[index].0 .0],
180                        other.sinks[index].0 .1,
181                    ),
182                    dyn_clone::clone_box(&*other.sinks[index].1),
183                )
184            });
185
186        let new_system: System<INPUTS, T> = System {
187            graph: self.graph,
188            layers: self.layers,
189            sources: self.sources,
190            sinks: new_sinks,
191        };
192
193        new_system
194    }

196    // Adds a filter to the system. Further references to this filter should be done using the returned uuid
197    pub fn add_filter(&mut self, filter: Box<dyn Filter>) -> NodeIndex<u32> {
198        trace!("[Graph] Adding filter {}", filter.get_name());
199        let nodeindex = self.graph.add_node(filter);
200        self.graph[nodeindex].set_index(nodeindex.index());
201        nodeindex
202    }

204    // Connects two filters together. This method connects the filter in the topologyu graph as well.
205    // Do no use this function to close a feedback loop. Use the connect_feedback method instead.
206    pub fn connect(
207        &mut self,
208        from: NodeIndex<u32>,
209        to: NodeIndex<u32>,
210        out_port: usize,
211        in_port: usize,
212    ) {
213        trace!(
214            "[Graph] Connecting {} (p: {}) to {} (p: {})",
215            self.graph[from].get_name(),
216            out_port,
217            self.graph[to].get_name(),
218            in_port
219        );
220        self.graph.add_edge(from, to, (out_port, in_port));
221    }

228    /// Connects a filter from the graph to a sink
229    pub fn connect_sink(&mut self, from: NodeIndex<u32>, sink: usize, out_port: usize) {
230        info!("Node {} (p: {}) -> Sink {}", from.index(), out_port, sink);
231        self.sinks[sink].0 = (from, out_port);

246    // Creates the execution layers by sorting the graph topologically.
247    #[allow(clippy::result_unit_err)]
248    pub fn compute(&mut self) -> Result<(), ()> {
249        // Makes the graph acyclic to be able to create a topology sort
250        let acyclic_graph = self.graph.filter_map(
251            |_index, node| Some(node),
252            |index, edge| {
253                if self
254                    .graph
255                    .edge_endpoints(index)
256                    .map(|(_, to)| self.graph[to].postponable())
257                    == Some(true)
258                {
259                    None
260                } else {
261                    Some(edge)
262                }
263            },
264        );
265
266        if let Ok(topo) = toposort(&acyclic_graph, None) {
267            for node in topo {
268                // TODO: Add same-layer ability (to run some filters in parallel)
269                // For each node in the topological order, push the node's index into the layers vector
270                // if the next node has no dependencies to the current node, push it to the current layer as well
271                // else push it to the next layer
272                self.layers.push(vec![node.index()])
273            }
274
275            Ok(())
276        } else {
277            Err(())
278        }
279    }

281    // Performs one full run of the system, running every filter once in an order such that data that entered the system this
282    // run, can exit it this run as well.
283    pub fn run(&mut self) {
284        self.sources.iter_mut().for_each(|(source, (desc, port))| {
285            let value = source.pull();
286            let filter = &mut self.graph[*desc];
287            filter.push(value, *port);
288        });
289
290        for layer in self.layers.iter() {
291            // TODO: Make this parallel
292            layer.iter().for_each(|f| {
293                let from_node_index = NodeIndex::new(*f);
294                let outputs = {
295                    let filter = &mut self.graph[from_node_index];
296                    filter.transform()
297                };
298
299                let neighbours: Vec<NodeIndex> = {
300                    self.graph
301                        .neighbors_directed(from_node_index, Direction::Outgoing)
302                        .collect()
303                };
304
305                for neighbour in neighbours {
306                    let edges: Vec<(usize, usize)> = {
307                        self.graph
308                            .edges_connecting(from_node_index, neighbour)
309                            .map(|e| *e.weight())
310                            .collect()
311                    };
312                    let neighbour_node = &mut self.graph[neighbour];
313
314                    edges.iter().for_each(|edge| {
315                        neighbour_node.push(outputs[edge.0], edge.1);
316                    });
317                }
318            });
319        }
320
321        // Makes the sinks pull from their connected nodes
322        self.sinks.iter_mut().for_each(|((node, port), sink)| {
323            let values = {
324                let node = &mut self.graph[*node];
325                node.transform()
326            };
327            sink.push(values[*port], 0);
328        });
329    }

331    /// Returns a sink pipe from the system. If the index is out of bounds, returns an error.
332    pub fn get_sink(&mut self, index: usize) -> Result<&mut Box<dyn Sink>, &str> {
333        self.sinks
334            .get_mut(index)
335            .map(|s| &mut s.1)
336            .ok_or("Index out of bounds")
337    }

339    /// Returns a mutable reference to a filter in the graph by its NodeIndex.
340    /// This allows direct access to filter-specific methods (like reset()) that aren't

342    pub fn get_filter_mut(&mut self, index: NodeIndex<u32>) -> Option<&mut Box<dyn Filter>> {
