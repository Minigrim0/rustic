36/// Reduce a collection of blocks arriving at the same port using the given mix strategy.
37fn mix_blocks(blocks: Vec<Block>, mode: &MixMode, block_size: usize) -> Block {
38    match blocks.len() {
39        0 => silent_block(block_size),
40        1 => blocks.into_iter().next().unwrap(),
41        _ => {
42            let count = blocks.len();
43            match mode {
44                MixMode::Sum | MixMode::Average => {
45                    let mut acc = silent_block(block_size);
46                    for block in blocks {
47                        for (af, bf) in acc.iter_mut().zip(block.iter()) {
48                            for ch in 0..CHANNELS {
49                                af[ch] += bf[ch];
50                            }
51                        }
52                    }
53                    if matches!(mode, MixMode::Average) {
54                        let inv = 1.0 / count as f32;
55                        acc.iter_mut()
56                            .for_each(|f| f.iter_mut().for_each(|s| *s *= inv));
57                    }
58                    acc
59                }
60                MixMode::Max => blocks
61                    .into_iter()
62                    .reduce(|a, b| {
63                        a.iter()
64                            .zip(b.iter())
65                            .map(|(fa, fb)| {
66                                let mut f = *fa;
67                                for ch in 0..CHANNELS {
68                                    f[ch] = fa[ch].max(fb[ch]);
69                                }
70                                f
71                            })
72                            .collect()
73                    })
74                    .unwrap(),
75                MixMode::Min => blocks
76                    .into_iter()
77                    .reduce(|a, b| {
78                        a.iter()
79                            .zip(b.iter())
80                            .map(|(fa, fb)| {
81                                let mut f = *fa;
82                                for ch in 0..CHANNELS {
83                                    f[ch] = fa[ch].min(fb[ch]);
84                                }
85                                f
86                            })
87                            .collect()
88                    })
89                    .unwrap(),
90            }
91        }
92    }
93}

95/// ## A Pipe & Filter system
96/// The system is composed of filters, sources and sinks.
97/// It is represented as a directed graph where the filters are the nodes.
98/// The Sources & Sinks are special nodes that have respectively only outgoing (source) or incoming (sink) edges.
99/// The edges represent the pipes between the filters.
100/// Some filters have special output properties. E.g. the delay filter's input pipe is ignored (postponed) when
101/// the topology sorting is done, in order to avoid cycles. A system with cycles must include a delay or similar filter
102/// to break the cycle.
103///
104/// ```rust
105/// use rustic::core::graph::System;
106/// use rustic::core::filters::prelude::Tremolo;
107///
108/// // A simple system with one input and one output
109/// let mut system = System::new();
110///
111/// // Adding a filter to the system
112/// let filter = Tremolo::new(20.0, 0.5, 44100.0);
113/// let filter_index = system.add_filter(Box::from(filter));
114/// ```
115#[derive(Debug, Clone)]
116#[allow(clippy::type_complexity)]
117pub struct System {
118    // The actual filter graph, from which the execution order is derived
119    // Each weight represents the port into which the filter is connected
120    graph: Graph<Box<dyn Filter>, (usize, usize)>,
121    // Each layer represents filters that can be run concurrently.
122    layers: Vec<Vec<usize>>,
123    // The sources of the system and the filters they are connected to.
124    // Each source may fan out to multiple (filter, port) pairs.
125    sources: Vec<(Box<dyn Source>, Vec<(NodeIndex<u32>, usize)>)>,
126    // The sinks of the system.
127    // The node index is the index of the filter that the sink is connected to
128    // The second usize is the port of the filter that the sink is connected to
129    sinks: Vec<((NodeIndex<u32>, usize), Box<dyn Sink>)>,
130    /// Direct source→sink wires that bypass the filter graph entirely.
131    /// Each entry is (source_index, sink_index).
132    source_sink_wires: Vec<(usize, usize)>,
133    /// Per-node mix strategy: how to combine multiple blocks arriving at the same input port.
134    /// Defaults to `MixMode::Sum` for nodes not present in this map.
135    mix_modes: HashMap<NodeIndex<u32>, MixMode>,
136    /// Live modulation wires: a source's block-mean drives a named parameter.
137    mod_wires: Vec<ModWire>,
138    /// Number of frames to produce per `run()` call
139    block_size: usize,
140}

185    /// Merges the two systems to create a new one. The graphs are merged following the given mapping from sinks to sources.
186    /// Sinks to sources links are replaced with a simple combinator filter. The amount of input in the second system
187    /// should match the amount of output in the first system.
188    #[allow(clippy::type_complexity)]
189    pub fn merge(
190        mut self,
191        other: System,
192        mapping: Vec<(usize, usize)>,
193    ) -> Result<System, AudioGraphError> {
194        if self.sinks.len() != other.sinks.len() {
195            log::error!("Trying to merge graphs with incompatible shapes");
196            return Err(AudioGraphError::InvalidMerging);
197        }
198
199        // Contains the mapping other graph -> new graph
200        let mut new_edge_map: HashMap<NodeIndex, NodeIndex> = HashMap::new();
201
202        log::trace!("Merging two graphs together");
203        for (from, to) in mapping.iter() {
204            let (graph_b_source_descendant_index, graph_b_node_port) = other.sources[*to]
205                .1
206                .first()
207                .copied()
208                .unwrap_or((NodeIndex::new(0), 0));
209            let source_descendant =
210                dyn_clone::clone_box(&*other.graph[graph_b_source_descendant_index]);
211
212            // Save the new index of the source descendant
213            let new_index = if let std::collections::hash_map::Entry::Vacant(e) =
214                new_edge_map.entry(graph_b_source_descendant_index)
215            {
216                let new_index = self.graph.add_node(source_descendant);
217                log::info!(
218                    "idx {} -> idx {}",
219                    graph_b_source_descendant_index.index(),
220                    new_index.index()
221                );
222                e.insert(new_index);
223                new_index
224            } else {
225                match new_edge_map.get(&graph_b_source_descendant_index) {
226                    Some(v) => *v,
227                    None => panic!("What ?"),
228                }
229            };
230
231            // Connect the sink's predecessors to the source's successors
232            let (sink_predecessor_id, sink_predecessor_port) = self.sinks[*from].0;
233            log::trace!(
234                "\tNode {} -> Sink {} & Source {} -> Node {} => Node {} -> Node {}",
235                sink_predecessor_id.index(),
236                from,
237                to,
238                graph_b_source_descendant_index.index(),
239                sink_predecessor_id.index(),
240                new_index.index()
241            );
242            self.graph.add_edge(
243                sink_predecessor_id,
244                new_index,
245                (sink_predecessor_port, graph_b_node_port),
246            );
247        }
248
249        // Go through all nodes in the other graph and add them to the new graph
250        for node_index in other.graph.node_indices() {
251            // Skip already added nodes
252            if new_edge_map.contains_key(&node_index) {
253                continue;
254            }
255
256            let node = dyn_clone::clone_box(&*other.graph[node_index]);
257            let new_index = self.graph.add_node(node);
258            new_edge_map.insert(node_index, new_index);
259        }
260
261        // Now that all nodes are added, connect the edges
262        for edge in other.graph.edge_indices() {
263            let (other_from, other_to) = other.graph.edge_endpoints(edge).unwrap();
264            let (from, to) = (new_edge_map[&other_from], new_edge_map[&other_to]);
265            log::info!(
266                "\tEdge ({}, {}) -> ({}, {})",
267                other_from.index(),
268                other_to.index(),
269                from.index(),
270                to.index()
271            );
272            let weight = other.graph[edge];
273            self.graph.add_edge(from, to, weight);
274        }
275
276        let new_sinks: Vec<_> = other
277            .sinks
278            .iter()
279            .map(|sink| {
280                (
281                    (new_edge_map[&sink.0.0], sink.0.1),
282                    dyn_clone::clone_box(&*sink.1),
283                )
284            })
285            .collect();
286
287        let new_system: System = System {
288            graph: self.graph,
289            layers: self.layers,
290            sources: self.sources,
291            sinks: new_sinks,
292            source_sink_wires: Vec::new(),
293            mix_modes: self.mix_modes,
294            mod_wires: Vec::new(),
295            block_size: self.block_size,
296        };
297
298        Ok(new_system)
299    }

412    /// Removes a source by index
413    pub fn remove_source(&mut self, index: usize) -> Option<Box<dyn Source>> {
414        if index < self.sources.len() {
415            let removed = self.sources.remove(index).0;
416            // Drop direct source→sink wires; shift remaining source indices
417            self.source_sink_wires.retain(|&(s, _)| s != index);
418            for (s, _) in self.source_sink_wires.iter_mut() {
419                if *s > index {
420                    *s -= 1;
421                }
422            }
423            // Drop mod wires involving the removed source; reindex surviving entries
424            self.mod_wires.retain(|w| w.from_source != index);
425            for w in self.mod_wires.iter_mut() {
426                if w.from_source > index {
427                    w.from_source -= 1;
428                }
429                if let ModTarget::Source(ref mut t) = w.target
430                    && *t > index
431                {
432                    *t -= 1;
433                }
434            }
435            Some(removed)
436        } else {
437            None
438        }
439    }

506    // Creates the execution layers by sorting the graph topologically.
507    #[allow(clippy::result_unit_err)]
508    pub fn compute(&mut self) -> Result<(), AudioGraphError> {
509        self.layers.clear();
510
511        // Makes the graph acyclic to be able to create a topology sort
512        let acyclic_graph = self.graph.filter_map(
513            |_index, node| Some(node),
514            |index, edge| {
515                if self
516                    .graph
517                    .edge_endpoints(index)
518                    .map(|(_, to)| self.graph[to].postponable())
519                    == Some(true)
520                {
521                    None
522                } else {
523                    Some(edge)
524                }
525            },
526        );
527
528        if let Ok(topo) = toposort(&acyclic_graph, None) {
529            for node in topo {
530                // TODO: Add same-layer ability (to run some filters in parallel)
531                self.layers.push(vec![node.index()])
532            }
533
534            Ok(())
535        } else {
536            Err(AudioGraphError::CycleDetected)
537        }
538    }

540    // Performs one full run of the system, running every filter once in an order such that data that entered the system this
541    // run, can exit it this run as well.
542    //
543    // Multiple connections to the same input port are accumulated and mixed using each filter's
544    // configured `MixMode` before the filter's `transform()` is called.
545    pub fn run(&mut self) {
546        let block_size = self.block_size;
547
548        // Accumulator: blocks pending delivery to (destination_filter, destination_port).
549        // All blocks for a port are collected here first, then mixed in one step.
550        let mut pending: HashMap<(NodeIndex<u32>, usize), Vec<Block>> = HashMap::new();
551
552        // Pull from all sources; enqueue blocks for all connected filters (fan-out)
553        let source_blocks: Vec<Block> = self
554            .sources
555            .iter_mut()
556            .enumerate()
557            .map(|(i, (source, connections))| {
558                let block = source.pull(block_size);
559                log::trace!("[system::run] source[{i}] active={}", source.is_active());
560                (block, connections.clone())
561            })
562            .collect::<Vec<_>>()
563            .into_iter()
564            .map(|(block, connections)| {
565                for (desc, port) in connections {
566                    if self.graph.node_weight(desc).is_some() {
567                        pending.entry((desc, port)).or_default().push(block.clone());
568                    }
569                }
570                block
571            })
572            .collect();
573
574        // Apply live modulation: drive target parameters from source block means.
575        // We collect the wires first to avoid borrowing self twice.
576        let mod_actions: Vec<(ModTarget, String, f32)> = self
577            .mod_wires
578            .iter()
579            .filter_map(|wire| {
580                let block = source_blocks.get(wire.from_source)?;
581                let mean = if block.is_empty() {
582                    0.0
583                } else {
584                    block.iter().map(|f| (f[0] + f[1]) * 0.5).sum::<f32>() / block.len() as f32
585                };
586                Some((wire.target.clone(), wire.param_name.clone(), mean))
587            })
588            .collect();
589        for (target, param_name, value) in mod_actions {
590            match target {
591                ModTarget::Source(idx) => {
592                    if let Some((src, _)) = self.sources.get_mut(idx) {
593                        src.set_parameter(&param_name, value);
594                    }
595                }
596                ModTarget::Filter(node_idx) => {
597                    if let Some(filter) = self.graph.node_weight_mut(node_idx) {
598                        filter.set_parameter(&param_name, value);
599                    }
600                }
601            }
602        }
603
604        // Process filters layer by layer; accumulate outputs for the next layer
605        let mut node_outputs: HashMap<NodeIndex<u32>, Vec<Block>> = HashMap::new();
606        for layer in self.layers.iter() {
607            for &f in layer.iter() {
608                let node_idx = NodeIndex::new(f);
609                let mix_mode = self.get_mix_mode(node_idx);
610
611                // Collect all ports that have pending blocks for this node
612                let ports: Vec<usize> = pending
613                    .keys()
614                    .filter(|(n, _)| *n == node_idx)
615                    .map(|(_, p)| *p)
616                    .collect();
617
618                // Mix and push each port's accumulated blocks
619                let filter = &mut self.graph[node_idx];
620                for port in ports {
621                    let blocks = pending.remove(&(node_idx, port)).unwrap_or_default();
622                    filter.push(mix_blocks(blocks, &mix_mode, block_size), port);
623                }
624
625                // Transform
626                let outputs = filter.transform();
627                // (filter borrow ends here; self.graph is available again)
628
629                // Enqueue outputs for downstream filters
630                let neighbours: Vec<NodeIndex> = self
631                    .graph
632                    .neighbors_directed(node_idx, Direction::Outgoing)
633                    .collect();
634                for neighbour in neighbours {
635                    let edges: Vec<(usize, usize)> = self
636                        .graph
637                        .edges_connecting(node_idx, neighbour)
638                        .map(|e| *e.weight())
639                        .collect();
640                    for (out_port, in_port) in edges {
641                        if let Some(block) = outputs.get(out_port) {
642                            pending
643                                .entry((neighbour, in_port))
644                                .or_default()
645                                .push(block.clone());
646                        }
647                    }
648                }
649
650                node_outputs.insert(node_idx, outputs);
651            }
652        }
653
654        // Push cached filter outputs to filter-connected sinks (no double-transform)
655        self.sinks
656            .iter_mut()
657            .enumerate()
658            .for_each(|(i, ((node, port), sink))| {
659                if let Some(outputs) = node_outputs.get(node) {
660                    if let Some(val) = outputs.get(*port) {
661                        log::trace!(
662                            "[system::run] sink[{i}] ← NodeIndex({}) port={port}",
663                            node.index()
664                        );
665                        sink.push(val.clone(), 0);
666                    } else {
667                        log::trace!(
668                            "[system::run] sink[{i}] ← NodeIndex({}) port={port} NOT FOUND",
669                            node.index()
670                        );
671                    }
672                } else {
673                    log::trace!(
674                        "[system::run] sink[{i}] ← NodeIndex({}) not in outputs",
675                        node.index()
676                    );
677                }
678            });
679
680        // Push source blocks to directly-wired sinks
681        for &(src_idx, sink_idx) in &self.source_sink_wires {
682            if let Some(block) = source_blocks.get(src_idx)
683                && let Some((_, sink)) = self.sinks.get_mut(sink_idx)
684            {
685                sink.push(block.clone(), 0);
686            }
687        }
688    }

705                self.sources.len()
706            );
707        }
708    }

780    /// Absorbs all filter nodes, edges, and sources from `other` into `self`,
781    /// remapping `NodeIndex`es to the new graph.
782    ///
783    /// Returns the remapped `NodeIndex` of the filter that was feeding `other`'s
784    /// first sink — the "output node" of the absorbed sub-graph — so the caller
785    /// can wire it into a master combinator or sink.
786    ///
787    /// `other`'s sinks are intentionally **not** imported; the caller is responsible
788    /// for providing a master sink and connecting to the returned output node.
789    pub fn absorb(&mut self, other: System) -> Result<NodeIndex<u32>, AudioGraphError> {
790        if other.sinks.is_empty() {
791            return Err(AudioGraphError::InvalidMerging);
792        }
793
794        // Record which filter node fed other's first sink before we consume other
795        let (other_output_node, _) = other.sinks[0].0;
796
797        // Import all filter nodes, building old-index → new-index mapping
798        let mut remap: HashMap<NodeIndex<u32>, NodeIndex<u32>> = HashMap::new();
799        for old_idx in other.graph.node_indices() {
800            let node = dyn_clone::clone_box(&*other.graph[old_idx]);
801            let new_idx = self.graph.add_node(node);
802            remap.insert(old_idx, new_idx);
803        }
804
805        // Re-add edges with remapped endpoints
806        for edge_idx in other.graph.edge_indices() {
807            let (from, to) = other.graph.edge_endpoints(edge_idx).unwrap();
808            let weight = other.graph[edge_idx];
809            self.graph.add_edge(remap[&from], remap[&to], weight);
810        }
811
812        // Transfer sources, remapping their connected filter NodeIndexes
813        for (source, connections) in other.sources {
814            let remapped: Vec<(NodeIndex<u32>, usize)> = connections
815                .into_iter()
816                .map(|(node_idx, port)| {
817                    let remapped_idx = remap.get(&node_idx).copied().unwrap_or(node_idx);
818                    (remapped_idx, port)
819                })
820                .collect();
821            self.sources.push((source, remapped));
822        }
823
824        // Return the remapped output node so the caller can connect it
825        remap
826            .get(&other_output_node)
827            .copied()
828            .ok_or(AudioGraphError::InvalidNode)
829    }
