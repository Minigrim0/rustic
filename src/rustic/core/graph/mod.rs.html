1//! Graphs are used to build audio pipelines for the application.
2//! They serve as metastructures for filters.
3//! Graphs can be used to create complex audio effects and processing chains.
4use dyn_clone::DynClone;
5use std::fmt;
6
7/// An element in the Audio pipeline
8/// Has a name and an index. Is able to connect to other elements
9pub trait AudioGraphElement: std::fmt::Debug {
10    /// Connects this element to another element.
11    /// The to element must implement the Entry trait.
12    fn get_name(&self) -> &str;
13    fn get_index(&self) -> usize;
14    fn set_index(&mut self, index: usize);
15}
16
17/// A trait for AudioGraphElements to allow other elements to push
18/// values into them.
19pub trait Entry: AudioGraphElement + DynClone {
20    /// Pushes a value into this element (sink or filter)
21    fn push(&mut self, value: f32, port: usize);
22}
23
24/// A trait for AudioGraphElements that allow neighbours in the graph to pull
25/// values from them. (Acts as a graph input)
26pub trait Source: AudioGraphElement {
27    /// Pull the next input from the source
28    fn pull(&mut self) -> f32;
29}
30
31/// A trait for AudioGraphElements that allow other parts of the
32/// code to consume values from them. (Acts as a graph output)
33pub trait Sink: Entry + AudioGraphElement {
34    /// Gets the values of the sink
35    fn consume(&mut self, amount: usize) -> Vec<f32>;
36    fn get_values(&self) -> Vec<f32>;
37    fn into_entry(self) -> Box<dyn Entry>;
38}
39
40/// A filter that can process data. Data should be pushed to the filter's input by either the preceding filter or a source.
41pub trait Filter: Entry + AudioGraphElement + fmt::Display + fmt::Debug {
42    /// Applies the filter's transformation to the input
43    /// Returns a tuple of the output and the indices of the elements that the filter is connected to
44    fn transform(&mut self) -> Vec<f32>;
45
46    /// Returns true if the filter's execution can be postponed to the end of the execution cycle of the graph.
47    /// An postonable element must be present in a cycle of the graph to avoid infinite looping.
48    /// E.g. a delay filter can be postponed if it lies within a feedback loop.
49    fn postponable(&self) -> bool;
50
51    /// Enables downcasting from trait object to concrete type.
52    /// This is required to access type-specific methods (like reset()) that aren't part of the Filter trait.
53    fn as_any_mut(&mut self) -> &mut dyn std::any::Any;
54}
