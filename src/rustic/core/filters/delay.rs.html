1use std::collections::VecDeque;

9/// Delays its input by a fixed amount of seconds.
10#[derive(Clone, Default)]
11#[cfg_attr(feature = "meta", derive(FilterMetaData))]
12pub struct DelayFilter {
13    #[cfg_attr(feature = "meta", filter_source)]
14    sources: [f32; 1],
15    #[cfg_attr(feature = "meta", filter_parameter(range, 0, 20.0, 0.5))]
16    delay_for: f32,
17    buffer: VecDeque<f32>,
18    index: usize,
19}
20
21impl DelayFilter {
22    pub fn new(sample_rate: f32, delay: f32) -> Self {
23        Self {
24            sources: [0.0],
25            delay_for: delay,
26            buffer: VecDeque::from(vec![0.0; (delay * sample_rate) as usize]),
27            index: 0,
28        }
29    }
30}
31
32impl Entry for DelayFilter {
33    fn push(&mut self, value: f32, port: usize) {
34        self.sources[port] = value;
35    }
36}
37
38impl fmt::Display for DelayFilter {
39    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
40        write!(f, "Delay Filter - {} samples", self.delay_for)
41    }
42}
43
44impl fmt::Debug for DelayFilter {
45    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
46        write!(
47            f,
48            "DelayFilter {{ delay_for: {}, index: {} }}",
49            self.delay_for, self.index
50        )
51    }
52}
53
54impl Filter for DelayFilter {
55    fn transform(&mut self) -> Vec<f32> {
56        let input = self.sources[0];
57        let output = self.buffer.pop_front().unwrap_or(0.0);
58
59        self.buffer.push_back(input);
60        vec![output]
61    }
62
63    fn postponable(&self) -> bool {
64        true
65    }
66
67    fn as_any_mut(&mut self) -> &mut dyn std::any::Any {
68        self
69    }
70}
71
72impl AudioGraphElement for DelayFilter {
73    fn get_name(&self) -> &str {
74        "Delay Filter"
75    }
76
77    fn get_index(&self) -> usize {
78        self.index
79    }
80
81    fn set_index(&mut self, index: usize) {
82        self.index = index;
