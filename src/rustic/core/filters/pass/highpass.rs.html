8/// High-pass filter using a first-order IIR filter
9#[derive(Clone, Debug, Default)]
10#[cfg_attr(feature = "meta", derive(FilterMetaData))]
11pub struct HighPassFilter {
12    #[cfg_attr(feature = "meta", filter_source)]
13    sources: [f32; 1],
14    #[cfg_attr(feature = "meta", filter_parameter(range, 0.0, 20000.0, 1000.0))]
15    cutoff_frequency: f32,
16    previous_output: f32,
17    index: usize,
18}
19
20impl HighPassFilter {
21    pub fn new(cutoff_frequency: f32) -> Self {
22        Self {
23            sources: [0.0],
24            cutoff_frequency,
25            previous_output: 0.0,
26            index: 0,
27        }
28    }
29}
30
31impl Entry for HighPassFilter {
32    fn push(&mut self, value: f32, port: usize) {
33        self.sources[port] = value;
34    }
35}
36
37impl fmt::Display for HighPassFilter {
38    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
39        write!(f, "High Pass Filter - {}Hz", self.cutoff_frequency)
40    }
41}
42
43impl Filter for HighPassFilter {
44    fn transform(&mut self) -> Vec<f32> {
45        let input = self.sources[0];
46        let alpha = 1.0 / (self.cutoff_frequency + 1.0);
47        let output = alpha * input + alpha * self.previous_output;
48        self.previous_output = output;
49        vec![output]
50    }
51
52    fn postponable(&self) -> bool {
53        false
54    }
55
56    fn as_any_mut(&mut self) -> &mut dyn std::any::Any {
57        self
58    }
59}
60
61impl AudioGraphElement for HighPassFilter {
62    fn get_name(&self) -> &str {
63        "High Pass Filter"
64    }
65
66    fn get_index(&self) -> usize {
67        self.index
68    }
69
70    fn set_index(&mut self, index: usize) {
71        self.index = index;
72    }
73}
