1//! AudioGraph — assembles all instruments into a single compiled `System`.
2//!
3//! `AudioGraph::compile()` calls `instrument.into_system()` for each slot,
4//! absorbs every sub-graph into one master graph, wires all outputs through a
5//! single `GainFilter` node at port 0 (the run-loop mixes them via `MixMode::Sum`),
6//! and returns the ready-to-run `System` for the render thread.
7
8use std::collections::HashMap;
9
10use crate::core::filters::prelude::GainFilter;
11use crate::core::graph::{AudioGraphError, AudioOutputSink, System};
12use crate::instruments::Instrument;
13
14/// A single instrument slot inside the audio graph.
15pub struct InstrumentSlot {
16    /// The instrument that will be converted to a sub-graph on compile.
17    pub instrument: Box<dyn Instrument>,
18    /// Optional per-instrument filter chain (merged after `into_system()`).
19    /// Currently unused — reserved for Phase 4.
20    pub filters: Option<System>,
21}
22
23/// Manages all instruments and compiles them into a single `System`.
24///
25/// After calling `compile()`, `source_map` records which source index inside
26/// the compiled `System` belongs to each instrument slot (by slot index).
27/// Use those indices with `System::start_note()` / `System::stop_note()`.
28#[derive(Default)]
29pub struct AudioGraph {
30    instruments: Vec<InstrumentSlot>,
31    /// Maps slot index → first source index in the most recent compiled System.
32    pub source_map: HashMap<usize, usize>,
33}
34
35impl AudioGraph {
36    pub fn new() -> Self {
37        Self::default()
38    }
39
40    /// Append an instrument and return its slot index.
41    pub fn add_instrument(&mut self, instrument: Box<dyn Instrument>) -> usize {
42        let idx = self.instruments.len();
43        self.instruments.push(InstrumentSlot {
44            instrument,
45            filters: None,
46        });
47        idx
48    }
49
50    /// Number of instrument slots.
51    pub fn len(&self) -> usize {
52        self.instruments.len()
53    }
54
55    pub fn is_empty(&self) -> bool {
56        self.instruments.is_empty()
57    }
58
59    /// Consume all instrument slots and compile them into a unified `System`.
60    ///
61    /// The returned `System` has one `AudioOutputSink` and is ready to be
62    /// swapped into the render thread. The `source_map` is updated so the
63    /// caller can route `NoteStart`/`NoteStop` to the correct source index.
64    ///
65    /// The `instruments` vec is emptied by this call; re-add instruments if
66    /// you need to compile again.
67    pub fn compile(&mut self) -> Result<System, AudioGraphError> {
68        if self.instruments.is_empty() {
69            return Ok(System::silent());
70        }
71
72        let slots: Vec<InstrumentSlot> = std::mem::take(&mut self.instruments);
73        let n = slots.len();
74
75        let mut main = System::new();
76        let mut output_nodes = Vec::with_capacity(n);
77
78        self.source_map.clear();
79
80        for (slot_idx, slot) in slots.into_iter().enumerate() {
81            let source_start = main.sources_len();
82
83            let inst_system = slot.instrument.into_system();
84            let output_node = main.absorb(inst_system)?;
85
86            // Every absorbed source up to source_start is this instrument's
87            let source_count = main.sources_len() - source_start;
88            if source_count > 0 {
89                self.source_map.insert(slot_idx, source_start);
90            }
91
92            output_nodes.push(output_node);
93        }
94
95        // Wire all instrument outputs into a passthrough gain node.
96        // All connect on the same port 0; run() accumulates and sums them automatically.
97        let mixer_node = main.add_filter(Box::new(GainFilter::new(1.0)));
98        for &out_node in output_nodes.iter() {
99            main.connect(out_node, mixer_node, 0, 0);
100        }
101
102        // Final sink
103        let sink = Box::new(AudioOutputSink::new());
104        let sink_idx = main.add_sink(sink);
105        main.connect_sink(mixer_node, sink_idx, 0);
106
107        main.compute()?;
108
109        Ok(main)
110    }
