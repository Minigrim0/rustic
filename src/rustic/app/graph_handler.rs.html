1//! Visual graph editor state and command handling.
2//!
3//! [`GraphData`] holds the user-built audio graph (nodes, connections, ID maps).
4//! [`handle_graph_command`] mutates it in response to [`GraphCommand`]s and
5//! hot-swaps the result into the render thread via a [`AudioMessage::Graph`] swap.
6
7use std::collections::HashMap;
8
9use petgraph::prelude::NodeIndex;

11use crate::app::commands::{GraphCommand, NodeKind};
12use crate::app::error::AppError;
13use crate::audio::{AudioMessage, GraphAudioMessage};
14use crate::core::generator::prelude::builder::*;
15use crate::core::generator::prelude::*;
16use crate::core::graph::{AudioOutputSink, Filter, ModTarget, SimpleSource, Source, System};

18/// State for the visual audio graph editor.
19#[derive(Default)]
20pub(crate) struct GraphData {
21    pub system: System,
22    pub filter_map: HashMap<u64, NodeIndex<u32>>,
23    pub source_map: HashMap<u64, usize>,
24    pub sink_map: HashMap<u64, usize>,
25}
26
27pub(crate) fn handle_graph_command(
28    cmd: GraphCommand,
29    gs: &mut GraphData,
30    sample_rate: f32,
31    message_tx: &crossbeam::channel::Sender<AudioMessage>,
32) -> Result<(), AppError> {
33    match cmd {
34        GraphCommand::AddNode {
35            id,
36            node_type,
37            kind,
38            ..
39        } => {
40            match kind {
41                NodeKind::Generator => {
42                    log::info!("Adding generator ({node_type}) to graph system");
43                    let source =
44                        create_source(&node_type, sample_rate).map_err(AppError::AudioError)?;
45                    let idx = gs.system.add_source(source);
46                    gs.source_map.insert(id, idx);
47                }
48                NodeKind::Filter => {
49                    log::info!("Adding filter ({node_type}) to graph system");
50                    let filter =
51                        create_filter(&node_type, sample_rate).map_err(AppError::AudioError)?;
52                    let idx = gs.system.add_filter(filter);
53                    gs.filter_map.insert(id, idx);
54                }
55                NodeKind::Sink => {
56                    log::info!("Adding sink to graph system");
57                    let sink = Box::new(AudioOutputSink::new());
58                    let idx = gs.system.add_sink(sink);
59                    gs.sink_map.insert(id, idx);
60                }
61            }
62            recompute_topology(gs)
63        }
64
65        GraphCommand::RemoveNode { id } => {
66            if let Some(idx) = gs.filter_map.remove(&id) {
67                gs.system.remove_filter(idx);
68                // NodeIndex-based map — petgraph handles index invalidation automatically
69            } else if let Some(idx) = gs.source_map.remove(&id) {
70                gs.system.remove_source(idx);
71                // Vec::remove shifts elements; update all stored source indices
72                for v in gs.source_map.values_mut() {
73                    if *v > idx {
74                        *v -= 1;
75                    }
76                }
77            } else if let Some(idx) = gs.sink_map.remove(&id) {
78                gs.system.remove_sink(idx);
79                // Vec::remove shifts elements; update all stored sink indices
80                for v in gs.sink_map.values_mut() {
81                    if *v > idx {
82                        *v -= 1;
83                    }
84                }
85            }
86            recompute_topology(gs)
87        }
88
89        GraphCommand::Connect {
90            from,
91            from_port,
92            to,
93            to_port,
94        } => {
95            let from_is_source = gs.source_map.contains_key(&from);
96            let to_is_sink = gs.sink_map.contains_key(&to);
97
98            if from_is_source && !to_is_sink {
99                let src_idx = gs.source_map[&from];
100                let filter_idx = gs.filter_map[&to];
101                gs.system.connect_source(src_idx, filter_idx, to_port);
102            } else if !from_is_source && to_is_sink {
103                let filter_idx = gs.filter_map[&from];
104                let sink_idx = gs.sink_map[&to];
105                gs.system.connect_sink(filter_idx, sink_idx, from_port);
106            } else if !from_is_source && !to_is_sink {
107                let from_idx = gs.filter_map[&from];
108                let to_idx = gs.filter_map[&to];
109                gs.system.connect(from_idx, to_idx, from_port, to_port);
110            } else {
111                // from_is_source && to_is_sink: direct wire, no filter in between
112                let src_idx = gs.source_map[&from];
113                let sink_idx = gs.sink_map[&to];
114                log::info!(
115                    "[graph] Connect: source {from}(idx={src_idx}) → sink {to}(idx={sink_idx}) direct wire"
116                );
117                gs.system.connect_source_to_sink(src_idx, sink_idx);
118            }
119            recompute_topology(gs)
120        }
121
122        GraphCommand::Disconnect { from, to } => {
123            let from_is_source = gs.source_map.contains_key(&from);
124            let to_is_sink = gs.sink_map.contains_key(&to);
125
126            if from_is_source && to_is_sink {
127                let src_idx = gs.source_map[&from];
128                let sink_idx = gs.sink_map[&to];
129                gs.system.disconnect_source_from_sink(src_idx, sink_idx);
130            } else if from_is_source && !to_is_sink {
131                if let (Some(&src_idx), Some(&filter_idx)) =
132                    (gs.source_map.get(&from), gs.filter_map.get(&to))
133                {
134                    gs.system.disconnect_source(src_idx, filter_idx);
135                }
136            } else if !from_is_source
137                && !to_is_sink
138                && let (Some(&from_idx), Some(&to_idx)) =
139                    (gs.filter_map.get(&from), gs.filter_map.get(&to))
140            {
141                let _ = gs.system.disconnect(from_idx, to_idx);
142            }
143            // filter→sink: sinks track a single (from, port) connection; the new connection
144            // from connect_sink() will overwrite it on the next Connect, so no explicit
145            // disconnect method is needed.
146            recompute_topology(gs)
147        }
148
149        GraphCommand::Modulate {
150            from,
151            to,
152            param_name,
153        } => {
154            if let Some(&src_idx) = gs.source_map.get(&from) {
155                let target = if let Some(&filter_idx) = gs.filter_map.get(&to) {
156                    Some(ModTarget::Filter(filter_idx))
157                } else if let Some(&tgt_src_idx) = gs.source_map.get(&to) {
158                    Some(ModTarget::Source(tgt_src_idx))
159                } else {
160                    None
161                };
162                if let Some(target) = target {
163                    gs.system
164                        .add_mod_wire(src_idx, target.clone(), param_name.clone());
165                    message_tx
166                        .send(AudioMessage::Graph(GraphAudioMessage::AddModulation {
167                            from_source: src_idx,
168                            target,
169                            param_name,
170                        }))
171                        .map_err(|_| AppError::ChannelClosed)?;
172                }
173            }
174            Ok(())
175        }
176
177        GraphCommand::Demodulate {
178            from,
179            to,
180            param_name,
181        } => {
182            if let Some(&src_idx) = gs.source_map.get(&from) {
183                let target = if let Some(&filter_idx) = gs.filter_map.get(&to) {
184                    Some(ModTarget::Filter(filter_idx))
185                } else if let Some(&tgt_src_idx) = gs.source_map.get(&to) {
186                    Some(ModTarget::Source(tgt_src_idx))
187                } else {
188                    None
189                };
190                if let Some(target) = target {
191                    gs.system.remove_mod_wire(src_idx, &target, &param_name);
192                    message_tx
193                        .send(AudioMessage::Graph(GraphAudioMessage::RemoveModulation {
194                            from_source: src_idx,
195                            target,
196                            param_name,
197                        }))
198                        .map_err(|_| AppError::ChannelClosed)?;
199                }
200            }
201            Ok(())
202        }
203
204        GraphCommand::SetParameter {
205            node_id,
206            param_name,
207            value,
208        } => {
209            if let Some(&idx) = gs.filter_map.get(&node_id) {
210                // Keep the command-thread topology copy in sync for mix_mode
211                if param_name == "mix_mode" {
212                    let mode = rustic_meta::MixMode::from_ordinal(value as usize);
213                    gs.system.set_mix_mode(idx, mode);
214                }
215                message_tx
216                    .send(AudioMessage::Graph(GraphAudioMessage::SetParameter {
217                        node_index: idx.index(),
218                        param_name,
219                        value,
220                    }))
221                    .map_err(|_| AppError::ChannelClosed)
222            } else if let Some(&src_idx) = gs.source_map.get(&node_id) {
223                // Keep the command-thread copy in sync
224                gs.system.set_source_parameter(src_idx, &param_name, value);
225                message_tx
226                    .send(AudioMessage::Graph(GraphAudioMessage::SetSourceParameter {
227                        source_index: src_idx,
228                        param_name,
229                        value,
230                    }))
231                    .map_err(|_| AppError::ChannelClosed)
232            } else {
233                Ok(())
234            }
235        }
236
237        GraphCommand::StartNode { id } => {
238            if let Some(&idx) = gs.source_map.get(&id) {
239                log::info!(
240                    "[graph] StartNode id={id} → source_index={idx}, sending StartSource to render thread"
241                );
242                message_tx
243                    .send(AudioMessage::Graph(GraphAudioMessage::StartSource {
244                        source_index: idx,
245                    }))
246                    .map_err(|_| AppError::ChannelClosed)
247            } else {
248                log::warn!(
249                    "[graph] StartNode id={id} → NOT found in source_map (known sources: {:?})",
250                    gs.source_map.keys().collect::<Vec<_>>()
251                );
252                Ok(())
253            }
254        }
255
256        GraphCommand::StopNode { id } => {
257            if let Some(&idx) = gs.source_map.get(&id) {
258                log::info!("[graph] StopNode id={id} → source_index={idx} (graceful stop)");
259                message_tx
260                    .send(AudioMessage::Graph(GraphAudioMessage::StopSource {
261                        source_index: idx,
262                    }))
263                    .map_err(|_| AppError::ChannelClosed)
264            } else {
265                Ok(())
266            }
267        }
268
269        GraphCommand::KillNode { id } => {
270            if let Some(&idx) = gs.source_map.get(&id) {
271                log::info!("[graph] KillNode id={id} → source_index={idx} (immediate kill)");
272                message_tx
273                    .send(AudioMessage::Graph(GraphAudioMessage::KillSource {
274                        source_index: idx,
275                    }))
276                    .map_err(|_| AppError::ChannelClosed)
277            } else {
278                Ok(())
279            }
280        }
281
282        GraphCommand::Compile => rebuild_and_swap(gs, message_tx),
283    }
284}

290/// Recompile the topology graph in the command-thread copy only.
291/// Does NOT push a Swap to the render thread — use `rebuild_and_swap` for that.
292fn recompute_topology(gs: &mut GraphData) -> Result<(), AppError> {
293    gs.system
294        .compute()
295        .map_err(|e| AppError::AudioError(format!("{:?}", e)))
296}
297
298/// Recompile the topology and immediately hot-swap it into the render thread.
299/// Only called by `GraphCommand::Compile`.
300fn rebuild_and_swap(
301    gs: &mut GraphData,
302    message_tx: &crossbeam::channel::Sender<AudioMessage>,
303) -> Result<(), AppError> {
304    gs.system
305        .compute()
306        .map_err(|e| AppError::AudioError(format!("{:?}", e)))?;
307    let cloned = gs.system.clone();
308    message_tx
309        .send(AudioMessage::Graph(GraphAudioMessage::Swap(cloned)))
310        .map_err(|_| AppError::ChannelClosed)
311}

313fn create_source(node_type: &str, sample_rate: f32) -> Result<Box<dyn Source>, String> {
314    let waveform: Waveform = node_type.into();
315    if let Waveform::Err(name) = waveform {
316        return Err(format!("Unknown generator type: {name}"));
317    }
318    // FrequencyRelation::Identity makes update_frequency() track the base frequency in
319    // real-time (1:1 ratio), so changing the "frequency" parameter while a note plays
320    // immediately affects the pitch. Without a relation the call is a no-op.
321    let generator: MultiToneGenerator = ToneGeneratorBuilder::new()
322        .waveform(waveform)
323        .frequency_relation(FrequencyRelation::Identity)
324        .build()
325        .into();
326    Ok(SimpleSource::new(generator, sample_rate).boxed())
327}

329fn create_filter(node_type: &str, sample_rate: f32) -> Result<Box<dyn Filter>, String> {
330    for entry in inventory::iter::<crate::meta::FilterRegistration>() {
331        let info = (entry.info)();
332        if info.name == node_type {
333            let mut filter = (entry.create)();
334            // Apply sample_rate to any filter that exposes it as a parameter.
335            // Filters without a "sample_rate" field will silently ignore this (logged at debug).
336            filter.set_parameter("sample_rate", sample_rate);
337            return Ok(filter);
338        }
339    }
340    Err(format!("Unknown filter type: {}", node_type))
341}
