9/// A staff is a single line in a music score. It has an associated instrument,
10/// contains multiple measures with their notes
11#[derive(Serialize, Deserialize, Default, Clone)]
12pub struct Staff {
13    instrument: usize, // Instrument index in the intrument map
14    measures: Vec<Measure>,
15    signature: TimeSignature,
16}
17
18impl Staff {
19    pub fn new(signature: &TimeSignature) -> Self {
20        Self {
21            signature: signature.clone(),
22            ..Default::default()
23        }
24    }
25
26    pub fn set_instrument(&mut self, instrument: usize) {
27        self.instrument = instrument;
28    }
29
30    pub fn get_instrument(&self) -> usize {
31        self.instrument
32    }
33
34    pub fn add_note(&mut self, note: Note) -> Result<(), String> {
35        let measure: &mut Measure =
36            if let Some(next_measure_position) = self.measures.iter().position(|m| !m.is_full()) {
37                &mut self.measures[next_measure_position]
38            } else {
39                self.measures.push(Measure::new(&self.signature));
40                self.measures.last_mut().unwrap()
41            };
42
43        let position = measure.current_index();
44        measure.add_note(position, note)
45    }
46
47    pub fn get_orderer_chords(&self) -> Vec<Chord> {
48        self.measures
49            .iter()
50            .flat_map(|m| m.get_orderer_chords())
51            .collect()
52    }
53
54    /// Calculates the total duration of this staff in ticks
55    pub fn total_duration(&self) -> usize {
56        let chords = self.get_orderer_chords();
57        chords.iter().map(|chord| chord.duration()).sum()
58    }
59}
