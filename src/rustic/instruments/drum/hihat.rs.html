20/// A HiHat instrument.
21/// It consists of six square wave sources connected to a combinator filter. The result is then passed through a resonant bandpass filter,
22/// before being shaped by an envelope generator.
23#[derive(Debug)]
24pub struct HiHat {
25    graph: System<1, 1>,
26    bandpass_filter_index: NodeIndex<u32>,
27    playing: bool,
28    time: f32,
29    #[cfg(debug_assertions)]
30    output_buffer: File,
31}
32
33impl HiHat {
34    pub fn new() -> Result<Self, String> {
35        let source = simple_source(
36            MultiToneGeneratorBuilder::new()
37                .add_generator(
38                    ToneGeneratorBuilder::new()
39                        .waveform(Waveform::Square)
40                        .frequency(123.0)
41                        .build(),
42                )
43                .add_generator(
44                    ToneGeneratorBuilder::new()
45                        .waveform(Waveform::Square)
46                        .frequency(150.0)
47                        .build(),
48                )
49                .add_generator(
50                    ToneGeneratorBuilder::new()
51                        .waveform(Waveform::Square)
52                        .frequency(180.0)
53                        .build(),
54                )
55                .add_generator(
56                    ToneGeneratorBuilder::new()
57                        .waveform(Waveform::Square)
58                        .frequency(219.0)
59                        .build(),
60                )
61                .add_generator(
62                    ToneGeneratorBuilder::new()
63                        .waveform(Waveform::Square)
64                        .frequency(240.0)
65                        .build(),
66                )
67                .add_generator(
68                    ToneGeneratorBuilder::new()
69                        .waveform(Waveform::Square)
70                        .frequency(261.0)
71                        .build(),
72                )
73                .amplitude_envelope(Some(Box::new(BezierSegment::new(
74                    1.0,
75                    0.0,
76                    0.5,
77                    (0.0, 0.0),
78                ))))
79                .mix_mode(MixMode::Average)
80                .build(),
81        );
82
83        let mut system = System::<1, 1>::new();
84        system.set_source(0, source);
85
86        let bandpass = system.add_filter(Box::from(ResonantBandpassFilter::new(
87            (10.0e3 + 400.0) / 2.0,
88            20.0,
89            44100.0,
90        )));
91
92        system.connect_source(0, bandpass, 0);
93
94        let sink: SimpleSink = SimpleSink::new();
95        system.set_sink(0, Box::from(sink));
96        system.connect_sink(bandpass, 0, 0);
97
98        system
99            .compute()
100            .map_err(|_| "Failed to compute".to_string())?;
101
102        match crate::app::prelude::FSConfig::debug_dir("HiHat", "hihat.viz") {
103            Ok(path) => {
104                if let Err(e) = system.save_to_file(&path) {
105                    log::warn!("Failed to save visualization: {}", e);
106                }
107            }
108            Err(_) => log::warn!("Failed to build path to save hihat graph"),
109        }
110
111        #[cfg(debug_assertions)]
112        let output_path =
113            crate::app::prelude::FSConfig::debug_dir("HiHat", "hihat_output.txt").unwrap();
114
115        Ok(Self {
116            graph: system,
117            bandpass_filter_index: bandpass,
118            playing: false,
119            time: 0.0,
120            #[cfg(debug_assertions)]
121            output_buffer: File::create(output_path).unwrap(),
122        })
123    }
124}

126impl Instrument for HiHat {
127    fn start_note(&mut self, _note: Note, _velocity: f32) {
128        log::trace!("Starting HiHat note");
129        self.playing = true;
130        self.time = 0.0;
131
132        // Reset the bandpass filter state to ensure clean retriggering.
133        // For percussive sounds, residual filter state from previous hits
134        // causes tonal artifacts and reduces transient clarity.
135        if let Some(filter_box) = self.graph.get_filter_mut(self.bandpass_filter_index) {
136            // Downcast to ResonantBandpassFilter to access its reset method
137            if let Some(bandpass) = filter_box
138                .as_any_mut()
139                .downcast_mut::<ResonantBandpassFilter>()
140            {
141                bandpass.reset();
142            } else {
143                log::warn!("Failed to downcast filter to ResonantBandpassFilter");
144            }
145        }
146    }
147
148    fn stop_note(&mut self, _note: Note) {
149        log::trace!("Stopping HiHat note");
150        self.playing = false;
151    }
152
153    fn get_output(&mut self) -> f32 {
154        let value = *self
155            .graph
156            .get_sink(0)
157            .unwrap()
158            .consume(1)
159            .first()
160            .unwrap_or(&0.0);
161        #[cfg(debug_assertions)]
162        {
163            // Check if the output buffer is empty
164            if self.output_buffer.metadata().unwrap().len() > 0 {
165                if let Err(e) = self.output_buffer.write(format!(" {}", value).as_bytes()) {
166                    log::warn!("Failed to write to output buffer: {}", e);
167                }
168            } else if let Err(e) = self.output_buffer.write(format!("{}", value).as_bytes()) {
169                log::warn!("Failed to write to output buffer: {}", e);
170            }
171        }
172        value
173    }
174
175    fn tick(&mut self) {
176        if self.playing {
177            self.graph.run();
178            self.time += 1.0 / 44100.0;
179        }
180    }
181}
