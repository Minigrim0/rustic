1use crate::core::envelope::prelude::{
2    ADSREnvelopeBuilder, BezierSegment, ConstantSegment, LinearSegment,
3};
4use crate::core::generator::prelude::{
5    builder::{MultiToneGeneratorBuilder, ToneGeneratorBuilder},
6    FrequencyRelation, MixMode, MultiToneGenerator, Waveform,
7};
8use crate::instruments::Instrument;
9use crate::Note;
10
11#[derive(Default, Debug)]
12pub struct Kick {
13    generator: MultiToneGenerator,
14    current_tick: u32,
15    output: f32,
16}
17
18impl Kick {
19    pub fn new() -> Self {
20        Self {
21            generator: MultiToneGeneratorBuilder::new()
22                .add_generator(
23                    ToneGeneratorBuilder::new()
24                        .waveform(Waveform::WhiteNoise)
25                        .frequency_relation(FrequencyRelation::Constant(1.0))
26                        .amplitude_envelope(Box::new(
27                            ADSREnvelopeBuilder::new()
28                                .attack(Box::new(BezierSegment::new(0.0, 0.1, 0.001, (0.0, 1.0))))
29                                .decay(Box::new(LinearSegment::new(0.1, 0.0, 0.1)))
30                                .release(Box::new(ConstantSegment::new(0.0, Some(0.0))))
31                                .build(),
32                        ))
33                        .build(),
34                )
35                .add_generator(
36                    ToneGeneratorBuilder::new()
37                        .waveform(Waveform::Sine)
38                        .frequency_relation(FrequencyRelation::Ratio(1.0))
39                        .amplitude_envelope(Box::new(ConstantSegment::new(1.0, None)))
40                        .build(),
41                )
42                .pitch_envelope(Some(Box::from(BezierSegment::new(
43                    1.0,
44                    0.5,
45                    0.3,
46                    (2.0, 0.2),
47                ))))
48                .mix_mode(MixMode::Sum)
49                .frequency(58.0)
50                .build(),
51            current_tick: 0,
52            output: 0.0,
53        }
54    }
55}
56
57impl Instrument for Kick {
58    fn start_note(&mut self, _note: Note, _velocity: f32) {
59        self.current_tick = 0;
60        self.generator.start();
61    }
62
63    fn stop_note(&mut self, _note: crate::Note) {
64        // The note will continue playing until completed
65        self.generator.stop();
66    }
67
68    fn get_output(&mut self) -> f32 {
69        self.output
70    }
71
72    fn tick(&mut self) {
73        self.output = self.generator.tick(1.0 / 44100.0);
74    }
75}
