Expand description

A reusable slider with an associated label

§Example

use frontend::widgets::LabeledSlider;

fn draw_ui(ui: &mut egui::Ui) {
    let mut value = 50.0;

    if LabeledSlider::new("Volume:", &mut value, 0.0..=100.0)
        .with_suffix("%")
        .show(ui)
        .changed()
    {
        // Handle the value change
        println!("New value: {}", value);
    }
}
